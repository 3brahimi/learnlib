017package de.learner.testsupport.it.learner;
018
019import org.testng.annotations.Test;
020
021import de.learner.testsupport.it.learner.internal.SingleExampleAllVariantsITSubCase;
022import de.learner.testsupport.it.learner.internal.SingleLearnerVariantITSubCase;
023
024/**
025 * Abstract integration test for a learning algorithm (or "learner").
026 * <p>
027 * A learner integration test tests the functionality of a learning algorithm against
028 * a well-defined set of example setups.
029 * <p>
030 * This class most probably does not need to be subclassed directly. Instead, extend
031 * one of the existing subclasses.
032 * 
033 * @author Malte Isberner
034 *
035 */
036public abstract class AbstractLearnerIT {
037
038        /*
039         * FIXME: Nested @Factory not working as expected, so we use this workaround ...
040         */
041        @Test
042        public void testAll() {
043                for(SingleExampleAllVariantsITSubCase<?, ?, ?> exampleSubCase : createExampleITCases()) {
044                        for(SingleLearnerVariantITSubCase<?, ?, ?> variantSubCase : exampleSubCase.createSingleVariantITCases()) {
045                                variantSubCase.testLearning();
046                        }
047                }
048        }
049        
050        /**
051         * Creates an array of per-example test cases for all learner variants.
052         * @return the array of test cases, one for each example 
053         */
054        public abstract SingleExampleAllVariantsITSubCase<?,?,?>[] createExampleITCases();
055
056}
