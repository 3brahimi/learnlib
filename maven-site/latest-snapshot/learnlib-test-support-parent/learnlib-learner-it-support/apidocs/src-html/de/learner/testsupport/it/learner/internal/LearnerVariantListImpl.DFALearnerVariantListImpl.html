017package de.learner.testsupport.it.learner.internal;
018
019import java.util.ArrayList;
020import java.util.List;
021
022import net.automatalib.automata.fsa.DFA;
023import net.automatalib.automata.transout.MealyMachine;
024import net.automatalib.words.Word;
025import de.learner.testsupport.it.learner.LearnerVariantList;
026import de.learnlib.api.LearningAlgorithm;
027import de.learnlib.mealy.MealyUtil;
028
029public class LearnerVariantListImpl<M,I,O> implements LearnerVariantList<M, I, O> {
030        
031        public static class DFALearnerVariantListImpl<I> extends LearnerVariantListImpl<DFA<?,I>,I,Boolean>
032                        implements DFALearnerVariantList<I> {}
033        public static class MealyLearnerVariantListImpl<I,O> extends LearnerVariantListImpl<MealyMachine<?,I,?,O>, I, Word<O>>
034                        implements MealyLearnerVariantList<I, O> {}
035        
036        public static class MealySymLearnerVariantListImpl<I,O> implements MealySymLearnerVariantList<I, O> {
037                
038                private final MealyLearnerVariantListImpl<I, O> mealyLearnerVariants
039                        = new MealyLearnerVariantListImpl<>();
040                
041                public MealyLearnerVariantListImpl<I, O> getMealyLearnerVariants() {
042                        return mealyLearnerVariants;
043                }
044
045                @Override
046                public void addLearnerVariant(
047                                String name,
048                                LearningAlgorithm<? extends MealyMachine<?, I, ?, O>, I, O> learner) {
049                        addLearnerVariant(name, learner, -1);
050                }
051
052                @Override
053                public void addLearnerVariant(
054                                String name,
055                                LearningAlgorithm<? extends MealyMachine<?, I, ?, O>, I, O> learner,
056                                int maxRounds) {
057                        mealyLearnerVariants.addLearnerVariant(name,
058                                        MealyUtil.wrapSymbolLearner(learner),
059                                        maxRounds);
060                }
061                
062        }
063        
064        private final List<LearnerVariant<M, I, O>> learnerVariants
065                = new ArrayList<>();
066
067        @Override
068        public void addLearnerVariant(String name,
069                        LearningAlgorithm<? extends M, I, O> learner) {
070                addLearnerVariant(name, learner, -1);
071        }
072
073        @Override
074        public void addLearnerVariant(String name,
075                        LearningAlgorithm<? extends M, I, O> learner, int maxRounds) {
076                LearnerVariant<M, I, O> variant
077                        = new LearnerVariant<>(name, learner, maxRounds);
078                learnerVariants.add(variant);
079        }
080                
081        
082        public List<? extends LearnerVariant<M,I,O>> getLearnerVariants() {
083                return learnerVariants;
084        }
085        
086}
