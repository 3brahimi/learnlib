017package de.learnlib.oracles;
018
019import javax.annotation.Nonnull;
020import javax.annotation.ParametersAreNonnullByDefault;
021
022import net.automatalib.words.Word;
023import net.automatalib.words.WordBuilder;
024import de.learnlib.api.MembershipOracle.MealyMembershipOracle;
025import de.learnlib.api.SUL;
026
027/**
028 * A wrapper around a system under learning (SUL).
029 * 
030 * @author falkhowar
031 */
032@ParametersAreNonnullByDefault
033public class SULOracle<I, O> extends AbstractSingleQueryOracle<I, Word<O>> implements MealyMembershipOracle<I,O> {
034
035        private final SUL<I, O> sul;
036
037        public SULOracle(SUL<I, O> sul) {
038                this.sul = sul;
039        }
040
041        @Override
042        @Nonnull
043        public Word<O> answerQuery(Word<I> prefix, Word<I> suffix) {
044                sul.pre();
045                // Prefix: Execute symbols, don't record output
046                for(I sym : prefix) {
047                        sul.step(sym);
048                }
049                
050                // Suffix: Execute symbols, outputs constitute output word
051                WordBuilder<O> wb = new WordBuilder<>(suffix.length());
052                for(I sym : suffix) {
053                        wb.add(sul.step(sym));
054                }
055                
056        sul.post();
057                return wb.toWord();
058        }
059
060}
