017package de.learnlib.examples.example2;

019import java.io.IOException;
020import java.io.Writer;
021import java.lang.reflect.Method;
022import java.util.ArrayDeque;
023import java.util.ArrayList;
024import java.util.Deque;
025import java.util.List;
026import java.util.Random;
027
028import net.automatalib.automata.transout.MealyMachine;
029import net.automatalib.commons.dotutil.DOT;
030import net.automatalib.util.graphs.dot.GraphDOT;
031import net.automatalib.words.Word;
032import de.learnlib.algorithms.lstargeneric.mealy.ExtensibleLStarMealyBuilder;
033import de.learnlib.api.EquivalenceOracle.MealyEquivalenceOracle;
034import de.learnlib.api.LearningAlgorithm.MealyLearner;
035import de.learnlib.api.SUL;
036import de.learnlib.cache.sul.SULCaches;
037import de.learnlib.drivers.reflect.AbstractMethodInput;
038import de.learnlib.drivers.reflect.AbstractMethodOutput;
039import de.learnlib.drivers.reflect.SimplePOJOTestDriver;
040import de.learnlib.eqtests.basic.mealy.RandomWalkEQOracle;
041import de.learnlib.experiments.Experiment.MealyExperiment;
042import de.learnlib.oracles.ResetCounterSUL;
043import de.learnlib.oracles.SULOracle;
044import de.learnlib.statistics.SimpleProfiler;
045import de.learnlib.statistics.StatisticSUL;

047/**
048 * This example shows how a model of a Java class can be learned using the SUL
049 * (system under learning) interfaces and the random walks equivalence test.
050 *
051 * @author falkhowar
052 */
053public class Example {
054
055    /*
056     * The BoundedStringQueue is the class of which we are going to 
057     * infer a model. It wraps an ordinary queue of Strings, limiting
058     * its size to MAX_SIZE (3). Once the queue is full, additional 
059     * offers will be ignored.
060     * <p>
061     * However, the implementation uses the underlying queue in a strange
062     * way as the model will reveal.
063     */
064    public static class BoundedStringQueue {
065
066        // capacity
067        public static final int MAX_SIZE = 3;
068        // storage
069        private Deque<String> data = new ArrayDeque<>(3);
070
071        // add a String to the queue if capacity allows
072        public void offer(String s) {
073            if (data.size() < MAX_SIZE) {
074                data.offerFirst(s);
075            }
076        }
077
078        // get next element from queue (null for empty queue)
079        public String poll() {
080            return data.poll();
081        }
082    }
083    
084    
085    public static void main(String[] args) throws NoSuchMethodException, IOException {
086
087        // instantiate test driver
088        SimplePOJOTestDriver driver = new SimplePOJOTestDriver(
089                BoundedStringQueue.class.getConstructor());
090                
091        // create learning alphabet
092        Method mOffer = BoundedStringQueue.class.getMethod(
093                "offer", new Class<?>[]{String.class});
094        Method mPoll = BoundedStringQueue.class.getMethod(
095                "poll", new Class<?>[]{});
096                
097        // offer
098        AbstractMethodInput offer_a = driver.addInput("offer_a", mOffer, "a");
099        AbstractMethodInput offer_b = driver.addInput("offer_b", mOffer, "b");
100
101        // poll
102        AbstractMethodInput poll = driver.addInput("poll", mPoll);
103
104        // oracle for counting queries wraps sul
105        StatisticSUL<AbstractMethodInput, AbstractMethodOutput> statisticSul = 
106                new ResetCounterSUL<>("membership queries", driver);
107        
108        SUL<AbstractMethodInput, AbstractMethodOutput> effectiveSul = statisticSul;
109        // use caching in order to avoid duplicate queries
110        effectiveSul = SULCaches.createCache(driver.getInputs(), effectiveSul);
111        
112        SULOracle<AbstractMethodInput, AbstractMethodOutput> mqOracle = new SULOracle<>(effectiveSul);
113
114        // create initial set of suffixes
115        List<Word<AbstractMethodInput>> suffixes = new ArrayList<>();
116        suffixes.add(Word.fromSymbols(offer_a));
117        suffixes.add(Word.fromSymbols(offer_b));
118        suffixes.add(Word.fromSymbols(poll));
119
120        // construct L* instance (almost classic Mealy version)
121        // almost: we use words (Word<String>) in cells of the table 
122        // instead of single outputs.
123        MealyLearner<AbstractMethodInput, AbstractMethodOutput> lstar
124                = new ExtensibleLStarMealyBuilder<AbstractMethodInput,AbstractMethodOutput>()
125                        .withAlphabet(driver.getInputs()) // input alphabet
126                        .withOracle(mqOracle)                     // membership oracle
127                        .create();
128                
129
130        // create random walks equivalence test
131        MealyEquivalenceOracle<AbstractMethodInput, AbstractMethodOutput> randomWalks =
132                new RandomWalkEQOracle<>(
133                0.05, // reset SUL w/ this probability before a step 
134                10000, // max steps (overall)
135                false, // reset step count after counterexample 
136                new Random(46346293), // make results reproducible 
137                driver // system under learning
138                );
139
140        // construct a learning experiment from
141        // the learning algorithm and the random walks test.
142        // The experiment will execute the main loop of
143        // active learning
144        MealyExperiment<AbstractMethodInput, AbstractMethodOutput> experiment =
145                new MealyExperiment<>(lstar, randomWalks, driver.getInputs());
146
147        // turn on time profiling
148        experiment.setProfile(true);
149
150        // enable logging of models
151        experiment.setLogModels(true);
152
153        // run experiment
154        experiment.run();
155
156        // get learned model
157        MealyMachine<?, AbstractMethodInput, ?, AbstractMethodOutput> result = 
158                experiment.getFinalHypothesis();
159
160        // report results
161        System.out.println("-------------------------------------------------------");
162
163        // profiling
164        System.out.println(SimpleProfiler.getResults());
165
166        // learning statistics
167        System.out.println(experiment.getRounds().getSummary());
168        System.out.println(statisticSul.getStatisticalData().getSummary());
169
170        // model statistics
171        System.out.println("States: " + result.size());
172        System.out.println("Sigma: " + driver.getInputs().size());
173
174        // show model
175        System.out.println();
176        System.out.println("Model: ");
177        
178        GraphDOT.write(result, driver.getInputs(), System.out); // may throw IOException!
179        Writer w = DOT.createDotWriter(true);
180        GraphDOT.write(result, driver.getInputs(), w);
181        w.close();
182
183        System.out.println("-------------------------------------------------------");
184
185    }
186}
