017package de.learnlib.algorithms.lstargeneric.mealy;
018
019
020import java.util.ArrayList;
021import java.util.Collection;
022import java.util.List;
023
024import net.automatalib.automata.concepts.SuffixOutput;
025import net.automatalib.automata.transout.MealyMachine;
026import net.automatalib.automata.transout.impl.compact.CompactMealy;
027import net.automatalib.automata.transout.impl.compact.CompactMealyTransition;
028import net.automatalib.words.Alphabet;
029import net.automatalib.words.Word;
030
031import com.github.misberner.buildergen.annotations.GenerateBuilder;
032
033import de.learnlib.algorithms.features.observationtable.OTLearner.OTLearnerMealy;
034import de.learnlib.algorithms.lstargeneric.ExtensibleAutomatonLStar;
035import de.learnlib.algorithms.lstargeneric.ce.ObservationTableCEXHandler;
036import de.learnlib.algorithms.lstargeneric.closing.ClosingStrategy;
037import de.learnlib.algorithms.lstargeneric.table.Row;
038import de.learnlib.api.MembershipOracle;
039import de.learnlib.oracles.DefaultQuery;
040
041public class ExtensibleLStarMealy<I, O> extends
042                ExtensibleAutomatonLStar<MealyMachine<?,I,?,O>, I, Word<O>, Integer, CompactMealyTransition<O>, Void, O, CompactMealy<I,O>>
043                implements OTLearnerMealy<I,O> {
044        
045        private final List<O> outputTable
046                = new ArrayList<O>();
047        
048
049        @GenerateBuilder(defaults = ExtensibleAutomatonLStar.BuilderDefaults.class)
050        public ExtensibleLStarMealy(Alphabet<I> alphabet,
051                        MembershipOracle<I, Word<O>> oracle,
052                        Collection<? extends Word<I>> initialSuffixes,
053                        ObservationTableCEXHandler<? super I, ? super Word<O>> cexHandler,
054                        ClosingStrategy<? super I, ? super Word<O>> closingStrategy) {
055                super(alphabet, oracle, new CompactMealy<I,O>(alphabet),
056                                LStarMealyUtil.ensureSuffixCompliancy(initialSuffixes, alphabet, cexHandler.needsConsistencyCheck()),
057                                cexHandler,
058                                closingStrategy);
059        }
060
061        @Override
062        protected void updateInternalHypothesis() {
063                updateOutputs();
064                super.updateInternalHypothesis();
065        }
066
067        @Override
068        protected Void stateProperty(Row<I> stateRow) {
069                return null;
070        }
071
072        @Override
073        protected O transitionProperty(Row<I> stateRow, int inputIdx) {
074                Row<I> transRow = stateRow.getSuccessor(inputIdx);
075                return outputTable.get(transRow.getRowId() - 1);
076        }
077
078        
079        protected void updateOutputs() {
080                int numOutputs = outputTable.size();
081                int numTransRows = table.numTotalRows() - 1;
082                
083                int newOutputs = numTransRows - numOutputs;
084                if(newOutputs == 0)
085                        return;
086                
087                List<DefaultQuery<I,Word<O>>> outputQueries
088                        = new ArrayList<DefaultQuery<I,Word<O>>>(numOutputs);
089                
090                for(int i = numOutputs+1; i <= numTransRows; i++) {
091                        Row<I> row = table.getRow(i);
092                        Word<I> rowPrefix = row.getPrefix();
093                        int prefixLen = rowPrefix.size();
094                        outputQueries.add(new DefaultQuery<I,Word<O>>(rowPrefix.prefix(prefixLen - 1),
095                                        rowPrefix.suffix(1)));
096                }
097                
098                oracle.processQueries(outputQueries);
099                
100                for(int i = 0; i < newOutputs; i++) {
101                        DefaultQuery<I,Word<O>> query = outputQueries.get(i);
102                        O outSym = query.getOutput().getSymbol(0);
103                        outputTable.add(outSym);
104                }
105        }
106
107        @Override
108        protected MealyMachine<?, I, ?, O> exposeInternalHypothesis() {
109                return internalHyp;
110        }
111        
112        @Override
113        protected SuffixOutput<I,Word<O>> hypothesisOutput() {
114                return internalHyp;
115        }
116
117}
