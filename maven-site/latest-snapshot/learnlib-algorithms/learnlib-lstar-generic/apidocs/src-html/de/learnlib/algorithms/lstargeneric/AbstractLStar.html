017package de.learnlib.algorithms.lstargeneric;
018
019import java.util.ArrayList;
020import java.util.Collection;
021import java.util.Collections;
022import java.util.List;
023import java.util.Objects;
024
025import net.automatalib.words.Alphabet;
026import net.automatalib.words.Word;
027import de.learnlib.algorithms.features.globalsuffixes.GlobalSuffixLearner;
028import de.learnlib.algorithms.features.observationtable.OTLearner;
029import de.learnlib.algorithms.lstargeneric.ce.ObservationTableCEXHandlers;
030import de.learnlib.algorithms.lstargeneric.table.Inconsistency;
031import de.learnlib.algorithms.lstargeneric.table.ObservationTable;
032import de.learnlib.algorithms.lstargeneric.table.Row;
033import de.learnlib.api.MembershipOracle;
034import de.learnlib.oracles.DefaultQuery;
035
036/**
037 * An abstract base class for L*-style algorithms.
038 * 
039 * This class implements basic management features (table, alphabet, oracle) and
040 * the main loop of alternating completeness and consistency checks. It does not take
041 * care of choosing how to initialize the table and hypothesis construction.
042 * 
043 * @author Malte Isberner <malte.isberner@gmail.com>
044 *
045 * @param <A> automaton class.
046 * @param <I> input symbol class.
047 * @param <O> output class.
048 */
049public abstract class AbstractLStar<A, I, O> implements OTLearner<A, I, O>, GlobalSuffixLearner<A, I, O> {
050        
051        protected final Alphabet<? extends I> alphabet;
052        protected final MembershipOracle<I, O> oracle;
053        protected final ObservationTable<I, O> table;
054
055        /**
056         * Constructor.
057         * @param alphabet the learning alphabet.
058         * @param oracle the membership oracle.
059         */
060        public AbstractLStar(Alphabet<I> alphabet, MembershipOracle<I,O> oracle) {
061                this.alphabet = alphabet;
062                this.oracle = oracle;
063                
064                this.table = new ObservationTable<>(alphabet);
065        }
066        
067        
068        /*
069         * (non-Javadoc)
070         * @see de.learnlib.api.LearningAlgorithm#start()
071         */
072        @Override
073        public void startLearning() {
074                Collection<? extends Word<I>> suffixes = initialSuffixes();
075                List<List<Row<I>>> initialUnclosed = table.initialize(suffixes, oracle);
076                
077                completeConsistentTable(initialUnclosed, false);
078        }
079
080        /*
081         * (non-Javadoc)
082         * @see de.learnlib.api.LearningAlgorithm#refineHypothesis(de.learnlib.api.Query)
083         */
084        @Override
085        public final boolean refineHypothesis(DefaultQuery<I, O> ceQuery) {
086                int oldDistinctRows = table.numDistinctRows();
087                doRefineHypothesis(ceQuery);
088                return (table.numDistinctRows() > oldDistinctRows);
089        }
090        
091        protected void doRefineHypothesis(DefaultQuery<I,O> ceQuery) {
092                List<List<Row<I>>> unclosed = incorporateCounterExample(ceQuery);
093                completeConsistentTable(unclosed, true);
094        }
095        
096        /**
097         * Iteratedly checks for unclosedness and inconsistencies in the table,
098         * and fixes any occurrences thereof. This process is repeated until the
099         * observation table is both closed and consistent. 
100         * @param unclosed the unclosed rows (equivalence classes) to start with.
101         */
102        protected boolean completeConsistentTable(List<List<Row<I>>> unclosed, boolean checkConsistency) {
103                boolean refined = false;
104                do {
105                        while(!unclosed.isEmpty()) {
106                                List<Row<I>> closingRows = selectClosingRows(unclosed);
107                                unclosed = table.toShortPrefixes(closingRows, oracle);
108                                refined = true;
109                        }
110                        
111                        
112                        if(checkConsistency) {
113                                Inconsistency<I,O> incons = null;
114                                
115                                do {
116                                        incons = table.findInconsistency();
117                                        if(incons != null) {
118                                                Word<I> newSuffix = analyzeInconsistency(incons);
119                                                unclosed = table.addSuffix(newSuffix, oracle);
120                                        }
121                                } while(unclosed.isEmpty() && (incons != null));
122                        }
123                } while(!unclosed.isEmpty());
124                
125                return refined;
126        }
127        
128        
129        /**
130         * Analyzes an inconsistency. This analysis consists in determining
131         * the column in which the two successor rows differ.
132         * @param incons the inconsistency description
133         * @return the suffix to add in order to fix the inconsistency
134         */
135        protected Word<I> analyzeInconsistency(Inconsistency<I,O> incons) {
136                int inputIdx = incons.getInputIndex();
137                
138                Row<I> succRow1 = incons.getFirstRow().getSuccessor(inputIdx);
139                Row<I> succRow2 = incons.getSecondRow().getSuccessor(inputIdx);
140                
141                int numSuffixes = table.numSuffixes();
142                
143                List<O> contents1 = table.rowContents(succRow1);
144                List<O> contents2 = table.rowContents(succRow2);
145                
146                for(int i = 0; i < numSuffixes; i++) {
147                        O val1 = contents1.get(i), val2 = contents2.get(i);
148                        if(!Objects.equals(val1, val2)) {
149                                I sym = alphabet.getSymbol(inputIdx);
150                                Word<I> suffix = table.getSuffixes().get(i);
151                                return suffix.prepend(sym);
152                        }
153                }
154                
155                throw new IllegalArgumentException("Bogus inconsistency");
156        }
157
158
159        /**
160         * Incorporates the information provided by a counterexample into
161         * the observation data structure.
162         * @param ce the query which contradicts the hypothesis
163         * @return the rows (equivalence classes) which became unclosed by
164         * adding the information. 
165         */
166        protected List<List<Row<I>>> incorporateCounterExample(DefaultQuery<I,O> ce) {
167                return ObservationTableCEXHandlers.handleClassicLStar(ce, table, oracle);
168        }
169        
170        /**
171         * This method selects a set of rows to use for closing the table.
172         * It receives as input a list of row lists, such that each (inner) list contains
173         * long prefix rows with (currently) identical contents, which have no matching
174         * short prefix row. The outer list is the list of all those equivalence classes.
175         * 
176         * @param unclosed a list of equivalence classes of unclosed rows.
177         * @return a list containing a representative row from each class to move
178         * to the short prefix part.
179         */
180        protected List<Row<I>> selectClosingRows(List<List<Row<I>>> unclosed) {
181                List<Row<I>> closingRows = new ArrayList<Row<I>>(unclosed.size());
182                
183                for(List<Row<I>> rowList : unclosed)
184                        closingRows.add(rowList.get(0));
185                
186                return closingRows;
187        }
188        
189        
190        /**
191         * Returns the list of initial suffixes which are used to initialize the table.
192         * @return the list of initial suffixes.
193         */
194        protected abstract Collection<? extends Word<I>> initialSuffixes();
195        
196        /*
197         * (non-Javadoc)
198         * @see de.learnlib.algorithms.features.GlobalSuffixLearner#getGlobalSuffixes()
199         */
200        @Override
201        public Collection<? extends Word<I>> getGlobalSuffixes() {
202                return Collections.unmodifiableCollection(table.getSuffixes());
203        }
204        
205        /*
206         * (non-Javadoc)
207         * @see de.learnlib.algorithms.features.GlobalSuffixLearner#addGlobalSuffixes(java.util.Collection)
208         */
209        @Override
210        public boolean addGlobalSuffixes(Collection<? extends Word<I>> newGlobalSuffixes) {
211                List<List<Row<I>>> unclosed = table.addSuffixes(newGlobalSuffixes, oracle);
212                if(unclosed.isEmpty()) {
213                        return false;
214                }
215                return completeConsistentTable(unclosed, false);
216        }
217        
218        /*
219         * (non-Javadoc)
220         * @see de.learnlib.algorithms.features.observationtable.OTLearner#getObservationTable()
221         */
222        @Override
223        public de.learnlib.algorithms.features.observationtable.ObservationTable<I, O> getObservationTable() {
224                return table.asStandardTable();
225        }
226}
