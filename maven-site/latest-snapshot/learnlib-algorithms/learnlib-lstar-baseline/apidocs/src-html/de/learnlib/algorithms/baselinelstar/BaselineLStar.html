041/**
042 * Implementation of the L* algorithm by Dana Angluin
043 *
044 * @param <I>
045 *              input symbol class.
046 */
047public class BaselineLStar<I> implements OTLearner<DFA<?, I>, I, Boolean>, GlobalSuffixLearnerDFA<I> {
048
049        @Nonnull
050        private final Alphabet<I> alphabet;
051
052        @Nonnull
053        private final MembershipOracle<I, Boolean> oracle;
054
055        @Nonnull
056        private final ObservationTable<I> observationTable;
057
058        private boolean startLearningAlreadyCalled;
059
060        /**
061         * Initializes a newly created baseline L* implementation. After this, the method
062         * {@link #startLearning()} may be called once.
063         *
064         * @param alphabet
065         *              The {@link Alphabet} to learn.
066         * @param oracle
067         *              The {@link MembershipOracle} which is used for membership queries.
068         */
069        public BaselineLStar(@Nonnull Alphabet<I> alphabet, @Nonnull MembershipOracle<I, Boolean> oracle) {
070                this.alphabet = alphabet;
071                this.oracle = oracle;
072                this.observationTable = new ObservationTable<>();
073
074                for (I alphabetSymbol : alphabet) {
075                        observationTable.addLongPrefix(Word.fromLetter(alphabetSymbol));
076                }
077        }
078
079        @Override
080        public void startLearning() {
081                if (startLearningAlreadyCalled) {
082                        throw new IllegalStateException("startLearning may only be called once!");
083                }
084
085                final List<Word<I>> allSuffixes = observationTable.getSuffixes();
086
087                processMembershipQueriesForStates(observationTable.getShortPrefixLabels(), allSuffixes);
088                processMembershipQueriesForStates(observationTable.getLongPrefixLabels(), allSuffixes);
089
090                makeTableClosedAndConsistent();
091
092                startLearningAlreadyCalled = true;
093        }
094
095        @Override
096        public boolean refineHypothesis(@Nonnull DefaultQuery<I, Boolean> ceQuery) {
097                if (!startLearningAlreadyCalled) {
098                        throw new IllegalStateException("Unable to refine hypothesis before first learn iteration!");
099                }
100
101                LinkedHashSet<Word<I>> prefixes = prefixesOfWordNotInStates(ceQuery.getInput());
102
103                for (Word<I> prefix : prefixes) {
104                        observationTable.addShortPrefix(prefix);
105                }
106
107                observationTable.removeShortPrefixesFromLongPrefixes();
108
109                LinkedHashSet<Word<I>> newCandidates = getNewCandidatesFromPrefixes(prefixes);
110
111                for (Word<I> candidate : newCandidates) {
112                        observationTable.addLongPrefix(candidate);
113                }
114
115                processMembershipQueriesForStates(prefixes, observationTable.getSuffixes());
116                processMembershipQueriesForStates(newCandidates, observationTable.getSuffixes());
117
118                makeTableClosedAndConsistent();
119
120                return true;
121        }
122
123        @Nonnull
124        private LinkedHashSet<Word<I>> prefixesOfWordNotInStates(@Nonnull Word<I> word) {
125                List<Word<I>> states = observationTable.getShortPrefixLabels();
126
127                LinkedHashSet<Word<I>> prefixes = new LinkedHashSet<>();
128                for (Word<I> prefix : word.prefixes(false)) {
129                        if (!states.contains(prefix)) {
130                                prefixes.add(prefix);
131                        }
132                }
133
134                return prefixes;
135        }
136
137        @Nonnull
138        private LinkedHashSet<Word<I>> getNewCandidatesFromPrefixes(@Nonnull LinkedHashSet<Word<I>> prefixes) {
139                LinkedHashSet<Word<I>> newCandidates = new LinkedHashSet<>();
140
141                for (Word<I> prefix : prefixes) {
142                        Set<Word<I>> possibleCandidates = appendAlphabetSymbolsToWord(prefix);
143                        for (Word<I> possibleCandidate :possibleCandidates) {
144                                if (!observationTable.getShortPrefixLabels().contains(possibleCandidate)) {
145                                        newCandidates.add(possibleCandidate);
146                                }
147                        }
148                }
149
150                return newCandidates;
151        }
152
153        /**
154         * Appends each symbol of the alphabet (with size m) to the given word (with size w),
155         * thus returning m words with a length of w+1.
156         *
157         * @param word
158         *      The {@link Word} to which the {@link Alphabet} is appended.
159         * @return
160         *      A set with the size of the alphabet, containing each time the word
161         *      appended with an alphabet symbol.
162         */
163        @Nonnull
164        private LinkedHashSet<Word<I>> appendAlphabetSymbolsToWord(@Nonnull Word<I> word) {
165                LinkedHashSet<Word<I>> newCandidates = new LinkedHashSet<>(alphabet.size());
166                for (I alphabetSymbol : alphabet) {
167                        Word<I> newCandidate = word.append(alphabetSymbol);
168                        newCandidates.add(newCandidate);
169                }
170                return newCandidates;
171        }
172
173        @Override
174        public DFA<?, I> getHypothesisModel() {
175                if (!startLearningAlreadyCalled) {
176                        throw new IllegalStateException("Unable to get hypothesis model before first learn iteration!");
177                }
178
179                FastDFA<I> automaton = new FastDFA<>(alphabet);
180                Map<List<Boolean>, FastDFAState> dfaStates = Maps.newHashMapWithExpectedSize(
181                                observationTable.getShortPrefixRows().size());
182
183                for (ObservationTableRow<I> stateRow : observationTable.getShortPrefixRows()) {
184                        if (dfaStates.containsKey(stateRow.getContents())) {
185                                continue;
186                        }
187
188                        FastDFAState dfaState;
189
190                        if (stateRow.getLabel().isEmpty()) {
191                                dfaState = automaton.addInitialState();
192                        }
193                        else {
194                                dfaState = automaton.addState();
195                        }
196
197                        Word<I> emptyWord = Word.epsilon();
198                        int positionOfEmptyWord = observationTable.getSuffixes().indexOf(emptyWord);
199                        dfaState.setAccepting(stateRow.getContents().get(positionOfEmptyWord));
200                        dfaStates.put(stateRow.getContents(), dfaState);
201                }
202
203                for (ObservationTableRow<I> stateRow : observationTable.getShortPrefixRows()) {
204                        FastDFAState dfaState = dfaStates.get(stateRow.getContents());
205                        for (I alphabetSymbol : alphabet) {
206                                Word<I> word = stateRow.getLabel().append(alphabetSymbol);
207
208                                final int index = alphabet.getSymbolIndex(alphabetSymbol);
209                                dfaState.setTransition(index, dfaStates.get(observationTable.getRowForPrefix(word).getContents()));
210                        }
211                }
212
213                return automaton;
214        }
215
216        /**
217         * After calling this method the observation table is both closed and consistent.
218         */
219        private void makeTableClosedAndConsistent() {
220                boolean closedAndConsistent = false;
221
222                while (!closedAndConsistent) {
223                        closedAndConsistent = true;
224
225                        if (!observationTable.isClosed()) {
226                                closedAndConsistent = false;
227                                closeTable();
228                        }
229
230                        if (!observationTable.isConsistentWithAlphabet(alphabet)) {
231                                closedAndConsistent = false;
232                                ensureConsistency();
233                        }
234                }
235        }
236
237        /**
238         * After calling this method the observation table is closed.
239         */
240        private void closeTable() {
241                Word<I> candidate = observationTable.findUnclosedState();
242
243                while (candidate != null) {
244                        observationTable.moveLongPrefixToShortPrefixes(candidate);
245
246                        LinkedHashSet<Word<I>> newCandidates = appendAlphabetSymbolsToWord(candidate);
247                        for (Word<I> newCandidate : newCandidates) {
248                                observationTable.addLongPrefix(newCandidate);
249                        }
250
251                        processMembershipQueriesForStates(newCandidates, observationTable.getSuffixes());
252
253                        candidate = observationTable.findUnclosedState();
254                }
255        }
256
257        /**
258         * After calling this method the observation table is consistent.
259         */
260        private void ensureConsistency() {
261                InconsistencyDataHolder<I> dataHolder = observationTable.findInconsistentSymbol(alphabet);
262
263                if (dataHolder == null) {
264                        // It seems like this method has been called without checking if table is inconsistent first
265                        return;
266                }
267
268                Word<I> witness = observationTable.determineWitnessForInconsistency(dataHolder);
269                Word<I> newSuffix = Word.fromSymbols(dataHolder.getDifferingSymbol()).concat(witness);
270                observationTable.addSuffix(newSuffix);
271
272                List<Word<I>> singleSuffixList = Collections.singletonList(newSuffix);
273
274                processMembershipQueriesForStates(observationTable.getShortPrefixLabels(), singleSuffixList);
275                processMembershipQueriesForStates(observationTable.getLongPrefixLabels(), singleSuffixList);
276        }
277
278        /**
279         * When new states are added to the observation table, this method fills the table values. For each
280         * given state it sends one membership query for each specified suffix symbol to the oracle of the
281         * form (state,symbol).
282         *
283         * @param states
284         *              The new states which should be evaluated.
285         * @param suffixes
286         *              The suffixes which are appended to the states before sending the resulting word to the oracle.
287         */
288        private void processMembershipQueriesForStates(@Nonnull Collection<Word<I>> states,
289                        @Nonnull Collection<? extends Word<I>> suffixes) {
290                List<DefaultQuery<I, Boolean>> queries = new ArrayList<>(states.size());
291                for (Word<I> label : states) {
292                        for (Word<I> suffix : suffixes) {
293                                queries.add(new DefaultQuery<I, Boolean>(label, suffix));
294                        }
295                }
296
297                oracle.processQueries(queries);
298
299                
300                for(DefaultQuery<I,Boolean> query : queries) {
301                        Word<I> state = query.getPrefix();
302                        Word<I> suffix = query.getSuffix();
303                        observationTable.addResult(state, suffix, query.getOutput());
304                }
305        }
306
307        @Nonnull
308        public String getStringRepresentationOfObservationTable() {
309                return ObservationTablePrinter.getPrintableStringRepresentation(observationTable);
310        }
311
312        @Override
313        @Nonnull
314        public Collection<? extends Word<I>> getGlobalSuffixes() {
315                return Collections.unmodifiableCollection(observationTable.getSuffixes());
316        }
317
318        @Override
319        public boolean addGlobalSuffixes(@Nonnull Collection<? extends Word<I>> newGlobalSuffixes) {
320                observationTable.getSuffixes().addAll(newGlobalSuffixes);
321
322                int numStatesOld = observationTable.getShortPrefixRows().size();
323
324                processMembershipQueriesForStates(observationTable.getShortPrefixLabels(), newGlobalSuffixes);
325                processMembershipQueriesForStates(observationTable.getLongPrefixLabels(), newGlobalSuffixes);
326
327                closeTable();
328
329                return (observationTable.getShortPrefixRows().size() != numStatesOld);
330        }
331
332        @Override
333        @Nonnull
334        public de.learnlib.algorithms.features.observationtable.ObservationTable<I, Boolean> getObservationTable() {
335                return observationTable;
336        }
