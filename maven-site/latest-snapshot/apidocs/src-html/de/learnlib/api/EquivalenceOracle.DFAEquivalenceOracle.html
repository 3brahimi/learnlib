017package de.learnlib.api;
018
019import java.util.Collection;
020
021import javax.annotation.Nullable;
022import javax.annotation.ParametersAreNonnullByDefault;
023
024import net.automatalib.automata.fsa.DFA;
025import net.automatalib.automata.transout.MealyMachine;
026import net.automatalib.words.Word;
027import de.learnlib.oracles.DefaultQuery;
028
029
030/**
031 * An equivalence oracle, which checks hypothesis automata against the (possibly unknown)
032 * system under learning (SUL).
033 * <p>
034 * Please note that equivalence oracles are implicitly connected to a SUL, there is no explicit
035 * references in terms of a {@link MembershipOracle} or such. However, this might be different
036 * in implementing classes.
037 * <p>
038 * <b>CAVEAT:</b> Equivalence oracles serve as an abstraction to tackle the (generally undecidable)
039 * problem of black-box equivalence testing. The contract imposed by this interface is that
040 * results returned by the {@link #findCounterExample(Object, Collection)} method are in fact
041 * counterexamples, <b>BUT</b> a <tt>null</tt> result signalling no counterexample was found
042 * does <b>not</b> mean that there can be none.
043 * 
044 * @author Maik Merten <maikmerten@googlemail.com>
045 * @author Malte Isberner <malte.isberner@gmail.com>
046 * 
047 * @param <A> automaton class this equivalence oracle works on
048 * @param <I> input symbol class
049 * @param <O> output class
050 */
051@ParametersAreNonnullByDefault
052public interface EquivalenceOracle<A, I, O> {
053        
054        /**
055         * A specialization of the {@link EquivalenceOracle} interface for a DFA learning scenario.
056         * 
057         * @author Malte Isberner <malte.isberner@gmail.com>
058         *
059         * @param <I> input symbol class
060         */
061        public static interface DFAEquivalenceOracle<I> extends EquivalenceOracle<DFA<?,I>,I,Boolean> {}
062        
063        /**
064         * A specialization of the {@link EquivalenceOracle} interface for a Mealy learning scenario.
065         * 
066         * @author Malte Isberner <malte.isberner@gmail.com>
067         *
068         * @param <I> input symbol class
069         * @param <O> output symbol class
070         */
071        public static interface MealyEquivalenceOracle<I,O> extends EquivalenceOracle<MealyMachine<?,I,?,O>,I,Word<O>> {}
072        
073        
074        /**
075         * Searches for a counterexample disproving the subjected hypothesis.
076         * A counterexample is query which, when performed on the SUL, yields a different output
077         * than what was predicted by the hypothesis. If no counterexample could be found (this does
078         * not necessarily mean that none exists), <code>null</code> is returned.
079         * 
080         * @param hypothesis the conjecture
081         * @param inputs the set of inputs to consider, this should be a subset of the input alphabet
082         * of the provided hypothesis
083         * @return a query exposing different behavior, or <tt>null</tt> if no counterexample
084         * could be found. In case a non-<tt>null</tt> value is returned, the output field
085         * in the {@link DefaultQuery} contains the SUL output for the respective query.
086         */
087        @Nullable
088        public DefaultQuery<I, O> findCounterExample(A hypothesis, Collection<? extends I> inputs);  
089        
090}
