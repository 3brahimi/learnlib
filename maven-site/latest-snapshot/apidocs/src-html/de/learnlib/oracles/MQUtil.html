017package de.learnlib.oracles;
018
019import java.util.Collection;
020import java.util.Collections;
021
022import javax.annotation.Nonnull;
023import javax.annotation.Nullable;
024import javax.annotation.ParametersAreNonnullByDefault;
025
026import net.automatalib.words.Word;
027import de.learnlib.api.MembershipOracle;
028import de.learnlib.api.Query;
029import de.learnlib.api.QueryAnswerer;
030
031@ParametersAreNonnullByDefault
032public abstract class MQUtil {
033        
034        @Nullable
035        public static <I,O> O output(MembershipOracle<I,O> oracle, Word<I> queryWord) {
036                return query(oracle, queryWord).getOutput();
037        }
038        
039        @Nullable
040        public static <I,O> O output(MembershipOracle<I,O> oracle, Word<I> prefix, Word<I> suffix) {
041                return query(oracle, prefix, suffix).getOutput();
042        }
043        
044        @Nonnull
045        public static <I,O> DefaultQuery<I,O> query(MembershipOracle<I,O> oracle, Word<I> prefix, Word<I> suffix) {
046                DefaultQuery<I,O> qry = new DefaultQuery<>(prefix, suffix);
047                oracle.processQueries(Collections.singleton(qry));
048                return qry;
049        }
050        
051        @Nonnull
052        public static <I,O> DefaultQuery<I,O> query(MembershipOracle<I,O> oracle, Word<I> queryWord) {
053                return query(oracle, Word.<I>epsilon(), queryWord);
054        }
055        
056        public static <I,O> void answerQueries(QueryAnswerer<I,O> answerer, Collection<? extends Query<I,O>> queries) {
057                for(Query<I,O> query : queries) {
058                        Word<I> prefix = query.getPrefix();
059                        Word<I> suffix = query.getSuffix();
060                        O answer = answerer.answerQuery(prefix, suffix);
061                        query.answer(answer);
062                }
063        }
064}
