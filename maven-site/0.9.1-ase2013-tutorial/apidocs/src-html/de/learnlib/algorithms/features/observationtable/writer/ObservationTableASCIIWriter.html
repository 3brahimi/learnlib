017package de.learnlib.algorithms.features.observationtable.writer;
018
019import java.io.IOException;
020import java.util.List;
021
022import net.automatalib.words.Word;
023
024import com.google.common.base.Function;
025
026import de.learnlib.algorithms.features.observationtable.ObservationTable;
027import de.learnlib.algorithms.features.observationtable.ObservationTable.Row;
028
029public class ObservationTableASCIIWriter<I,O> extends AbstractObservationTableWriter<I,O> {
030        
031        private boolean rowSeparators;
032        
033        public ObservationTableASCIIWriter(
034                        Function<? super Word<? extends I>,? extends String> wordToString,
035                        Function<? super O,? extends String> outputToString,
036                        boolean rowSeparators) {
037                super(wordToString, outputToString);
038                this.rowSeparators = rowSeparators;
039        }
040        
041        public ObservationTableASCIIWriter(boolean rowSeparators) {
042                this.rowSeparators = rowSeparators;
043        }
044        
045        public ObservationTableASCIIWriter() {
046                this(true);
047        }
048        
049        public void setRowSeparators(boolean rowSeparators) {
050                this.rowSeparators = rowSeparators;
051        }
052        
053        
054        @Override
055        public void write(ObservationTable<? extends I,? extends O> table, Appendable out) throws IOException {
056                List<? extends Word<? extends I>> suffixes = table.getSuffixes();
057                int numSuffixes = suffixes.size();
058                
059                int[] colWidth = new int[numSuffixes + 1];
060                
061                int i = 1;
062                for(Word<? extends I> suffix : suffixes) {
063                        colWidth[i++] = wordToString(suffix).length();
064                }
065                
066                for(Row<? extends I,? extends O> row : table.getAllRows()) {
067                        int thisWidth = wordToString(row.getLabel()).length();
068                        if(thisWidth > colWidth[0]) {
069                                colWidth[0] = thisWidth;
070                        }
071                        
072                        i = 1;
073                        for(O value : row) {
074                                thisWidth = outputToString(value).length();
075                                if(thisWidth > colWidth[i]) {
076                                        colWidth[i] = thisWidth;
077                                }
078                                i++;
079                        }
080                }
081                
082                
083                appendSeparatorRow(out, '=', colWidth);
084                String[] content = new String[numSuffixes + 1];
085                
086                // Header
087                content[0] = "";
088                i = 1;
089                for(Word<? extends I> suffix : suffixes) {
090                        content[i++] = wordToString(suffix);
091                }
092                appendContentRow(out, content, colWidth);
093                appendSeparatorRow(out, '=', colWidth);
094                
095                boolean first = true;
096                for(Row<? extends I,? extends O> spRow : table.getShortPrefixRows()) {
097                        if(first) {
098                                first = false;
099                        }
100                        else if(rowSeparators) {
101                                appendSeparatorRow(out, '-', colWidth);
102                        }
103                        content[0] = wordToString(spRow.getLabel());
104                        i = 1;
105                        for(O value : spRow) {
106                                content[i++] = outputToString(value);
107                        }
108                        appendContentRow(out, content, colWidth);
109                }
110                
111                appendSeparatorRow(out, '=', colWidth);
112                
113                first = true;
114                for(Row<? extends I,? extends O> lpRow : table.getLongPrefixRows()) {
115                        if(first) {
116                                first = false;
117                        }
118                        else if(rowSeparators) {
119                                appendSeparatorRow(out, '-', colWidth);
120                        }
121                        content[0] = wordToString(lpRow.getLabel());
122                        i = 1;
123                        for(O value : lpRow.getContents()) {
124                                content[i++] = outputToString(value);
125                        }
126                        appendContentRow(out, content, colWidth);
127                }
128                
129                appendSeparatorRow(out, '=', colWidth);
130        }
131        
132        
133        private static void appendSeparatorRow(Appendable a, char sepChar, int[] colWidth) throws IOException {
134                a.append('+').append(sepChar);
135                appendRepeated(a, sepChar, colWidth[0]);
136                for(int i = 1; i < colWidth.length; i++) {
137                        a.append(sepChar).append('+').append(sepChar);
138                        appendRepeated(a, sepChar, colWidth[i]);
139                }
140                a.append(sepChar).append("+\n");
141        }
142        
143        private static void appendContentRow(Appendable a, String[] content, int[] colWidth) throws IOException {
144                a.append("| ");
145                appendRightPadded(a, content[0], colWidth[0]);
146                for(int i = 1; i < content.length; i++) {
147                        a.append(" | ");
148                        appendRightPadded(a, content[i], colWidth[i]);
149                }
150                a.append(" |\n");
151        }
152        
153        private static void appendRightPadded(Appendable a, String string, int width) throws IOException {
154                a.append(string);
155                appendRepeated(a, ' ', width - string.length());
156        }
157        
158        private static void appendRepeated(Appendable a, char c, int count) throws IOException {
159                for(int i = 0; i < count; i++) {
160                        a.append(c);
161                }
162        }
163        
164}
