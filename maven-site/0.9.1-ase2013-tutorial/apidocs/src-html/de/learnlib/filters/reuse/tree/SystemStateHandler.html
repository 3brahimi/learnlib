017package de.learnlib.filters.reuse.tree;
018
019import de.learnlib.filters.reuse.ReuseOracle.ReuseOracleBuilder;
020
021/**
022 * A implementation of this interface that is set to the {@link ReuseTree} 
023 * (see {@link ReuseOracleBuilder#withSystemStateHandler(SystemStateHandler)}) 
024 * will be informed about all removed system states whenever
025 * {@link ReuseTree#disposeSystemstates()} gets called.
026 * <p>
027 * The objective of this handler is that clearing system states from the reuse
028 * tree may also be resulting in cleaning up the SUL by e.g. perform tasks like
029 * removing persisted entities from a database.
030 * <p>
031 * Please note that the normal removal of system states (by sifting them down in
032 * the reuse tree by executing only suffixes of a query) is not be seen as a
033 * disposing.
034 * 
035 * @author Oliver Bauer <oliver.bauer@tu-dortmund.de>
036 * 
037 * @param <S> system state class
038 */
039public interface SystemStateHandler<S> {
040        /**
041         * The system state S will be removed from the {@link ReuseTree}.
042         *
043         * @param state
044         *      The state to remove.
045         */
046        void dispose(S state);
047}
