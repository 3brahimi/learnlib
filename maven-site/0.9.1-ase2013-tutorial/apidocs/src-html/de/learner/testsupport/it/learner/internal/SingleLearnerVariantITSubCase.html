017package de.learner.testsupport.it.learner.internal;
018
019import net.automatalib.automata.UniversalDeterministicAutomaton;
020import net.automatalib.automata.concepts.SuffixOutput;
021import net.automatalib.util.automata.Automata;
022import net.automatalib.words.Alphabet;
023
024import org.testng.Assert;
025import org.testng.ITest;
026import org.testng.annotations.Test;
027
028import de.learnlib.api.EquivalenceOracle;
029import de.learnlib.api.LearningAlgorithm;
030import de.learnlib.eqtests.basic.SimulatorEQOracle;
031import de.learnlib.examples.LearningExample;
032import de.learnlib.oracles.DefaultQuery;
033
034public final class SingleLearnerVariantITSubCase<I,O,
035        M extends UniversalDeterministicAutomaton<?, I, ?, ?, ?> & SuffixOutput<I,O>> implements ITest {
036        
037        private final LearnerVariant<? extends M, I, O> variant;
038        private final LearningExample<I, O, ? extends M> example;
039        
040        public SingleLearnerVariantITSubCase(LearnerVariant<? extends M,I,O> variant, LearningExample<I,O,? extends M> example) {
041                this.variant = variant;
042                this.example = example;
043        }
044
045        @Override
046        public String getTestName() {
047                return variant.getLearner().getClass().getSimpleName() + "[" + variant.getName() + "]/" + example.getClass().getSimpleName();
048        }
049        
050        @Test
051        public void testLearning() {
052                System.out.println("Running learner integration test " + getTestName());
053                LearningAlgorithm<? extends M,I,O> learner
054                        = variant.getLearner();
055                
056                Alphabet<I> alphabet = example.getAlphabet();
057                
058                int maxRounds = variant.getMaxRounds();
059                if(maxRounds < 0) {
060                        maxRounds = example.getReferenceAutomaton().size();
061                }
062
063                EquivalenceOracle<? super M, I, O> eqOracle
064                        = new SimulatorEQOracle<I,O>(example.getReferenceAutomaton());
065                
066                learner.startLearning();
067                
068                int roundCounter = 0;
069                DefaultQuery<I, O> ceQuery;
070                
071                while((ceQuery = eqOracle.findCounterExample(learner.getHypothesisModel(), alphabet)) != null) {
072                        roundCounter++;
073                        if(roundCounter > maxRounds) {
074                                Assert.fail("Learning took too many rounds (> " + maxRounds + ")");
075                        }
076                        
077                        boolean refined = learner.refineHypothesis(ceQuery);
078                        Assert.assertTrue(refined, "Real counterexample did not refine hypothesis");
079                }
080                
081                Assert.assertNull(
082                                Automata.findSeparatingWord(example.getReferenceAutomaton(), learner.getHypothesisModel(), alphabet),
083                                "Final hypothesis does not match reference automaton");
084        }
085
086
087}
