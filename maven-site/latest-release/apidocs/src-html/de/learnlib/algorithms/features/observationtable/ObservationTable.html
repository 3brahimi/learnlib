017package de.learnlib.algorithms.features.observationtable;
018
019import java.util.Collection;
020import java.util.List;
021
022import javax.annotation.Nonnegative;
023import javax.annotation.Nonnull;
024import javax.annotation.Nullable;
025import javax.annotation.ParametersAreNonnullByDefault;
026import javax.annotation.Signed;
027
028import net.automatalib.words.Word;
029
030/**
031 * An observation table is a common method for learning algorithms to store organize
032 * their observations. This interface defines a generic API for interacting with
033 * homogeneously interacting with observation tables regardless of their implementation.
034 * <p>
035 * Instances implementing this interface can be obtained from learning algorithms implementing
036 * the {@link ObservationTableFeature observation table feature} (or {@link OTLearner}s.
037 * <p>
038 * Basically, an observation table is a two-dimensional table, where both rows and columns
039 * are indexed by {@link Word}s. The row indices are called <i>prefixes</i>, whereas
040 * the column indexes are referred to as <i>suffixes</i>. The table is further vertically
041 * divided into two halves: the prefixes in the upper half are referred to as
042 * <i>short prefixes</i> (these usually correspond to states in learned hypothesis automata),
043 * whereas the prefixes in the lower half are referred to as <i>long prefixes</i>. Long
044 * prefixes must be one-letter extensions of short prefixes; they refer to transitions
045 * in the hypothesis automaton. We refer to rows as <i>short prefix rows</i> or <i>long prefix
046 * row</i>, depending on whether they occur in the upper or lower half of the table respectively. 
047 * <p>
048 * The cells of the table are filled with observations for a given prefix and suffix combination.
049 * The type of observations is generic and can be specified using the type parameter {@code O}.
050 * <p>
051 * There are two important properties of observation tables, which usually have to be satisfied
052 * in order to be able to generate an automaton from an observation table: it must be both
053 * <i>closed</i> and <i>consistent</i>.
054 * <p>
055 * In a <b>closed</b> observation table, the contents of <i>each</i> long prefix row equal the
056 * contents of at least one short prefix rows. <b>Consistency</b>, on the other hand, is satisfied
057 * when for every two distinct short prefix rows, all rows indexed by one-letter extensions
058 * of the corresponding prefix with any input symbol also have the same content.
059 * 
060 * @author Malte Isberner
061 *
062 * @param <I> input symbol type
063 * @param <O> observation (output) type
064 */
065@ParametersAreNonnullByDefault
066public interface ObservationTable<I, O> {
067        
068        /**
069         * Used to indicate that no distinguishing suffix exists in
070         * {@link #findDistinguishingSuffixIndex(Inconsistency)} and
071         * {@link #findDistinguishingSuffixIndex(Row, Row)}.
072         */
073        public static final int NO_DISTINGUISHING_SUFFIX = -1;
074        
075        /**
076         * A single row in the observation table.
077         * 
078         * @author Malte Isberner
079         *
080         * @param <I> input symbol type 
081         * @param <O> observation type
082         */
083        public static interface Row<I,O> extends Iterable<O> {
084                /**
085                 * Retrieves the label of this row.
086                 * @return the label of this row
087                 */
088                @Nonnull
089                public Word<I> getLabel();
090                
091                /**
092                 * Retrieves whether this row is a short or a long prefix row.
093                 * @return {@code true} if this row is a short prefix row, {@code false}
094                 * otherwise.
095                 */
096                public boolean isShortPrefixRow();
097                
098                /**
099                 * Retrieves a list of the cell contents in this row.
100                 * @return the cell contents in this row
101                 */
102                @Nonnull
103                public List<? extends O> getContents();
104                
105                /**
106                 * Retrieves the size (length) of this row.
107                 * @return the size of this row
108                 */
109                public int size();
110                
111                /**
112                 * Retrieves the cell content at the given index.
113                 * @param index the index
114                 * @return the cell content at the given index
115                 * @throws IndexOutOfBoundsException if the given index is invalid, i.e., is less than
116                 * {@code 0} or greater than or equal to {@code size()}
117                 */
118                @Nullable
119                public O getCellContent(@Nonnegative int index) throws IndexOutOfBoundsException;
120        }
121        
122        
123        /**
124         * Representation of an inconsistency in the observation table.
125         * <p>
126         * An inconsistency is represented by two short prefix rows with
127         * matching contents, and an input symbol such that the rows indexed
128         * by one-letter extensions have differing contents.
129         * 
130         * @author Malte Isberner
131         *
132         * @param <I> input symbol type
133         * @param <O> observation type
134         */
135        public static interface Inconsistency<I,O> {
136                /**
137                 * Retrieves the first (short prefix) row constituting the inconsistency.
138                 * @return the first row
139                 */
140                @Nonnull
141                public Row<I,O> getFirstRow();
142                
143                /**
144                 * Retrieves the second (short prefix) row constituting the inconsistency
145                 * @return the second row
146                 */
147                @Nonnull
148                public Row<I,O> getSecondRow();
149                
150                /**
151                 * Retrieves the symbol for which's one-letter extensions the corresponding
152                 * rows have different contents. 
153                 * @return the symbol
154                 */
155                @Nullable
156                public I getSymbol();
157        }
158        
159        /**
160         * Retrieves the short prefixes in the table. The prefixes are returned
161         * in no specified order.
162         * @return the short prefixes in the table
163         */
164        @Nonnull
165        public Collection<? extends Word<I>> getShortPrefixes();
166        
167        /**
168         * Retrieves the long prefixes in the table. The prefixes are returned
169         * in no specified order.
170         * @return the long prefixes in the table
171         */
172        @Nonnull
173        public Collection<? extends Word<I>> getLongPrefixes();
174        
175        /**
176         * Retrieves all prefixes (short and long) in the table. The prefixes are
177         * returned in no specified order.
178         * @return all prefixes in the table
179         */
180        @Nonnull
181        public Collection<? extends Word<I>> getAllPrefixes();
182        
183        /**
184         * Retrieves all suffixes in the table.
185         * @return all suffixes in the table
186         */
187        @Nonnull
188        public List<? extends Word<I>> getSuffixes();
189        
190        /**
191         * Retrieves a suffix by its (column) index.
192         * 
193         * @param index
194         * @return
195         * @throws IndexOutOfBoundsException
196         */
197        @Nonnull
198        public Word<I> getSuffix(@Nonnegative int index) throws IndexOutOfBoundsException;
199        
200        @Nonnull
201        public Collection<? extends Row<I,O>> getShortPrefixRows();
202        @Nonnull
203        public Collection<? extends Row<I,O>> getLongPrefixRows();
204        @Nonnull
205        public Collection<? extends Row<I,O>> getAllRows();
206        
207        @Nonnull
208        public Row<I,O> getRow(Word<I> prefix) throws NoSuchRowException;                           
209        
210        @Nullable
211        public Row<I,O> getSuccessorRow(Row<I,O> spRow, @Nullable I symbol) throws InvalidRowException;
212        
213        
214        public boolean isClosed();
215        @Nullable
216        public Row<I,O> findUnclosedRow();
217        
218        public boolean isConsistent(Collection<? extends I> inputs);
219        @Nullable
220        public Inconsistency<I,O> findInconsistency(Collection<? extends I> inputs);
221        
222        /**
223         * 
224         * @param inconsistency
225         * @return the suffix (column) index where the contents of the rows differ, or
226         * {@code #NO_DISTINGUISHING_SUFFIX} if the contents of the rows are equal.
227         * @throws NoSuchRowException if the 
228         */
229        @Signed
230        public int findDistinguishingSuffixIndex(Inconsistency<I,O> inconsistency) throws NoSuchRowException, InvalidRowException;
231        
232        /**
233         * 
234         * @param inconsistency
235         * @return
236         * @throws NoSuchRowException
237         */
238        @Nullable
239        public Word<I> findDistinguishingSuffix(Inconsistency<I,O> inconsistency) throws NoSuchRowException, InvalidRowException;
240        
241        /**
242         * 
243         * @param firstRow the first row
244         * @param secondRow the second row
245         * @return the suffix (column) index where the contents of the rows differ, or
246         * {@code #NO_DISTINGUISHING_SUFFIX} if the contents of the rows are equal.  
247         * @throws InvalidRowException if the rows do not belong to this observation table
248         */
249        @Signed
250        public int findDistinguishingSuffixIndex(Row<I,O> firstRow, Row<I,O> secondRow) throws InvalidRowException;
251        
252        /**
253         * 
254         * @param firstRow the first row
255         * @param secondRow the second row
256         * @return the suffix distinguishing the contents of the two rows
257         * @throws InvalidRowException if the rows do not belong to this observation table
258         */
259        @Nullable
260        public Word<I> findDistinguishingSuffix(Row<I,O> firstRow, Row<I,O> secondRow) throws InvalidRowException;    
261}
