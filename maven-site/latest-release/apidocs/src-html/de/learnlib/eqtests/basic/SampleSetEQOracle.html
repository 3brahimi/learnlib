017package de.learnlib.eqtests.basic;
018
019import java.util.ArrayList;
020import java.util.Arrays;
021import java.util.Collection;
022import java.util.Iterator;
023import java.util.LinkedList;
024import java.util.List;
025import java.util.Objects;
026
027import net.automatalib.automata.concepts.SuffixOutput;
028import net.automatalib.words.Word;
029import de.learnlib.api.EquivalenceOracle;
030import de.learnlib.api.MembershipOracle;
031import de.learnlib.api.Query;
032import de.learnlib.oracles.DefaultQuery;
033
034
035/**
036 * An equivalence oracle that tests a hypothesis against a fixed set of sample queries.
037 * <p>
038 * Sample queries are provided through one of the {@code add(...)} or {@code addAll(...)}
039 * methods of this class. A query consists of an <i>input word</i> (split into a <i>prefix</i>
040 * and a <i>suffix</i>), and an expected <i>output</i>. During an equivalence query, for each
041 * of those queries if the respective actual suffix output of the hypothesis equals the expected
042 * output.
043 * <p>
044 * This oracle will always repeatedly test queries from the sample set if they turned out
045 * to be counterexamples. However, the oracle can be configured to remove queries from the
046 * sample set if they did not serve as counterexamples.
047 * 
048 * @author Malte Isberner <malte.isberner@gmail.com>
049 *
050 * @param <I> input symbol type
051 * @param <O> output type
052 */
053public class SampleSetEQOracle<I, O> implements EquivalenceOracle<SuffixOutput<I, O>, I, O>{
054        
055        private final boolean removeUnsuccessful;
056        private final List<DefaultQuery<I,O>> testQueries;
057        
058        /**
059         * Constructor. Initializes the oracle with an empty sample set.
060         * 
061         * @param removeUnsuccessful if set to {@code true}, queries will be removed
062         * from the sample set if they did not reveal a counterexample. Otherwise,
063         * all queries from the sample set will always be tested upon each invocation
064         * of {@link #findCounterExample(SuffixOutput, Collection)}.
065         */
066        public SampleSetEQOracle(boolean removeUnsuccessful) {
067                this.removeUnsuccessful = removeUnsuccessful;
068                if(!removeUnsuccessful) {
069                        testQueries = new ArrayList<>();
070                }
071                else {
072                        testQueries = new LinkedList<>(); // for O(1) removal of elements
073                }
074        }
075        
076        /**
077         * Adds a query word along with its expected output to the sample set.
078         * 
079         * @param input the input word
080         * @param expectedOutput the expected output for this word
081         * @return {@code this}, to enable chained {@code add} or {@code addAll} calls
082         */
083        public SampleSetEQOracle<I, O> add(Word<I> input, O expectedOutput) {
084                testQueries.add(new DefaultQuery<>(input, expectedOutput));
085                return this;
086        }
087        
088        /**
089         * Adds several query words to the sample set. The expected output is determined by
090         * means of the specified membership oracle.
091         * 
092         * @param oracle the membership oracle used to determine expected outputs
093         * @param words the words to be added to the sample set
094         * @return {@code this}, to enable chained {@code add} or {@code addAll} calls
095         */
096        @SafeVarargs
097        public final SampleSetEQOracle<I, O> addAll(MembershipOracle<I,O> oracle, Word<I> ...words) {
098                return addAll(oracle, Arrays.asList(words));
099        }
100        
101        /**
102         * Adds queries to the sample set. These must be {@link DefaultQuery}s, which allow
103         * for retrieving the corresponding (expected) output.
104         * 
105         * @param newTestQueries the queries to add to the sample set
106         * @return {@code this}, to enable chained {@code add} or {@code addAll} calls
107         */
108        @SafeVarargs
109        public final SampleSetEQOracle<I,O> addAll(DefaultQuery<I,O>... newTestQueries) {
110                return addAll(Arrays.asList(newTestQueries));
111        }
112        
113        /**
114         * Adds queries to the sample set. These must be {@link DefaultQuery}s, which allow
115         * for retrieving the corresponding (expected) output.
116         * 
117         * @param newTestQueries the queries to add to the sample set
118         * @return {@code this}, to enable chained {@code add} or {@code addAll} calls
119         */
120        public SampleSetEQOracle<I,O> addAll(Collection<? extends DefaultQuery<I,O>> newTestQueries) {
121                testQueries.addAll(newTestQueries);
122                return this;
123        }
124        
125        /**
126         * Adds words to the sample set. The expected output is determined by means
127         * of the specified membership oracle.
128         * 
129         * @param oracle the membership oracle used to determine the expected output
130         * @param words the words to add
131         * @return {@code this}, to enable chained {@code add} or {@code addAll} calls
132         */
133        public SampleSetEQOracle<I, O> addAll(MembershipOracle<I,O> oracle, Collection<? extends Word<I>> words) {
134                if(words.isEmpty()) {
135                        return this;
136                }
137                List<DefaultQuery<I,O>> newQueries = new ArrayList<>(words.size());
138                for(Word<I> w : words) {
139                        newQueries.add(new DefaultQuery<I,O>(w));
140                }
141                oracle.processQueries(newQueries);
142                
143                testQueries.addAll(newQueries);
144                return this;
145        }
146        
147        /*
148         * (non-Javadoc)
149         * @see de.learnlib.api.EquivalenceOracle#findCounterExample(java.lang.Object, java.util.Collection)
150         */
151        @Override
152        public DefaultQuery<I, O> findCounterExample(SuffixOutput<I, O> hypothesis,
153                        Collection<? extends I> inputs) {
154                Iterator<DefaultQuery<I,O>> queryIt = testQueries.iterator();
155                
156                while(queryIt.hasNext()) {
157                        DefaultQuery<I,O> query = queryIt.next();
158                        
159                        if(checkInputs(query, inputs)) {
160                                if(!test(query, hypothesis)) {
161                                        return query;
162                                }
163                                else if(removeUnsuccessful) {
164                                        queryIt.remove();
165                                }
166                        }
167                }
168                return null;
169        }
170        
171        
172        /**
173         * Tests if the suffix output of the given hypothesis matches the expected output stored
174         * in the query.
175         * @param query the query, containing the expected output
176         * @param hypOutput the suffix output portion of the hypothesis
177         * @return {@code true} if the suffix output by {@code hypOutput} matches the expected
178         * output stored in {@code query}, {@code false} otherwise. 
179         */
180        private static <I,O> boolean test(DefaultQuery<I, O> query, SuffixOutput<I,O> hypOutput) {
181                O hypOut = hypOutput.computeSuffixOutput(query.getPrefix(), query.getSuffix());
182                
183                return Objects.equals(hypOut, query.getOutput());
184        }
185        
186        /**
187         * Tests if the input word of the given {@link Query} consists entirely of symbols
188         * in {@code inputs}.
189         * @param query the query to test
190         * @param inputs the set of allowed inputs
191         * @return {@code true} if the input word of {@code query} consists entirely of symbols
192         * in {@code inputs}, {@code false} otherwise
193         */
194        private static <I> boolean checkInputs(Query<I,?> query, Collection<? extends I> inputs) {
195                for(I sym : query.getPrefix()) {
196                        if(!inputs.contains(sym)) {
197                                return false;
198                        }
199                }
200                for(I sym : query.getSuffix()) {
201                        if(!inputs.contains(sym)) {
202                                return false;
203                        }
204                }
205                return true;
206        }
207        
208}
