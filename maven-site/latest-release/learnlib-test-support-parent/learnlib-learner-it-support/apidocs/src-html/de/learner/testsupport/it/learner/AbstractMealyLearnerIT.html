017package de.learner.testsupport.it.learner;

019import java.util.List;
020
021import net.automatalib.automata.transout.MealyMachine;
022import net.automatalib.words.Alphabet;
023import net.automatalib.words.Word;
024import de.learner.testsupport.it.learner.LearnerVariantList.MealyLearnerVariantList;
025import de.learner.testsupport.it.learner.internal.LearnerVariantListImpl.MealyLearnerVariantListImpl;
026import de.learner.testsupport.it.learner.internal.SingleExampleAllVariantsITSubCase;
027import de.learnlib.api.MembershipOracle.MealyMembershipOracle;
028import de.learnlib.examples.LearningExample.MealyLearningExample;
029import de.learnlib.examples.LearningExamples;
030import de.learnlib.oracles.SimulatorOracle.MealySimulatorOracle;
031
032/**
033 * Abstract integration test for Mealy machine learning algorithms.
034 * <p>
035 * Mealy machine learning algorithms tested by this integration test are expected to
036 * assume membership queries yield the full output word corresponding to the suffix
037 * part of the query. If the learning algorithm only expects the last symbol as
038 * output, use {@link AbstractMealySymLearnerIT}.
039 *  
040 * @author Malte Isberner
041 *
042 */
043public abstract class AbstractMealyLearnerIT extends AbstractLearnerIT {
044
045        
046        // @Factory FIXME
047        @Override
048        public SingleExampleAllVariantsITSubCase<?,?,?>[] createExampleITCases() {
049                List<? extends MealyLearningExample<?,?>> examples = LearningExamples.createMealyExamples();
050                
051                SingleExampleAllVariantsITSubCase<?,?,?>[] result = new SingleExampleAllVariantsITSubCase[examples.size()];
052                int i = 0;
053                for(MealyLearningExample<?,?> example : examples) {
054                        result[i++] = createAllVariantsITCase(example);
055                }
056                
057                return result;
058        }
059        
060        private <I,O>
061        SingleExampleAllVariantsITSubCase<I, Word<O>, MealyMachine<?,I,?,O>> createAllVariantsITCase(MealyLearningExample<I,O> example) {
062                Alphabet<I> alphabet = example.getAlphabet();
063                MealyMembershipOracle<I,O> mqOracle
064                        = new MealySimulatorOracle<>(example.getReferenceAutomaton());
065                MealyLearnerVariantListImpl<I,O> variants = new MealyLearnerVariantListImpl<>();
066                addLearnerVariants(alphabet, mqOracle, variants);
067                
068                return new SingleExampleAllVariantsITSubCase<>(example, variants);
069        }
070        
071        /**
072         * Adds, for a given setup, all the variants of the Mealy machine learner to be tested
073         * to the specified {@link LearnerVariantList variant list}.
074         * 
075         * @param alphabet the input alphabet
076         * @param mqOracle the membership oracle
077         * @param variants list to add the learner variants to
078         */
079        protected abstract <I,O> void addLearnerVariants(
080                        Alphabet<I> alphabet,
081                        MealyMembershipOracle<I,O> mqOracle,
082                        MealyLearnerVariantList<I,O> variants);
083}
