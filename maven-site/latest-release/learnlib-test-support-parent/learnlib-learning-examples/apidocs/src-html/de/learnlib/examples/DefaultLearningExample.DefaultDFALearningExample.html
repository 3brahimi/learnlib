017package de.learnlib.examples;
018import net.automatalib.automata.UniversalAutomaton;
019import net.automatalib.automata.concepts.InputAlphabetHolder;
020import net.automatalib.automata.concepts.SuffixOutput;
021import net.automatalib.automata.fsa.DFA;
022import net.automatalib.automata.transout.MealyMachine;
023import net.automatalib.words.Alphabet;
024import net.automatalib.words.Word;
025
026/**
027 * A {@link LearningExample learning example} that directly stores the alphabet and the reference automaton
028 * in its fields.
029 * 
030 * @author Malte Isberner
031 *
032 * @param <I> input symbol type
033 * @param <O> output type
034 * @param <A> automaton type
035 */
036public class DefaultLearningExample<I, O, A extends UniversalAutomaton<?, I, ?, ?, ?> & SuffixOutput<I, O>>
037                implements LearningExample<I, O, A> {
038        
039        public static class DefaultDFALearningExample<I> extends DefaultLearningExample<I,Boolean,DFA<?,I>>
040                implements DFALearningExample<I> {
041                public DefaultDFALearningExample(Alphabet<I> alphabet,
042                                DFA<?, I> referenceAutomaton) {
043                        super(alphabet, referenceAutomaton);
044                }
045                public <A extends DFA<?,I> & InputAlphabetHolder<I>>
046                DefaultDFALearningExample(A automaton) {
047                        this(automaton.getInputAlphabet(), automaton);
048                }
049        }
050        
051        public static class DefaultMealyLearningExample<I,O> extends DefaultLearningExample<I,Word<O>,MealyMachine<?,I,?,O>>
052                implements MealyLearningExample<I,O> {
053                public DefaultMealyLearningExample(Alphabet<I> alphabet,
054                                MealyMachine<?, I, ?, O> referenceAutomaton) {
055                        super(alphabet, referenceAutomaton);
056                }
057                public <A extends MealyMachine<?,I,?,O> & InputAlphabetHolder<I>>
058                DefaultMealyLearningExample(A automaton) {
059                        this(automaton.getInputAlphabet(), automaton);
060                }
061        }
062        
063        private final Alphabet<I> alphabet;
064        private final A referenceAutomaton;
065
066        public DefaultLearningExample(Alphabet<I> alphabet, A referenceAutomaton) {
067                this.alphabet = alphabet;
068                this.referenceAutomaton = referenceAutomaton;
069        }
070        
071
072        @Override
073        public A getReferenceAutomaton() {
074                return referenceAutomaton;
075        }
076
077        @Override
078        public Alphabet<I> getAlphabet() {
079                return alphabet;
080        }
081
082}
