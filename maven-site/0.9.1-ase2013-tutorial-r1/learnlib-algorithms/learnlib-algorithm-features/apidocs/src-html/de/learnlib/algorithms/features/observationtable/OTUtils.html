017package de.learnlib.algorithms.features.observationtable;
018
019import java.awt.Desktop;
020import java.awt.HeadlessException;
021import java.io.BufferedWriter;
022import java.io.File;
023import java.io.FileWriter;
024import java.io.IOException;
025import java.util.concurrent.FutureTask;
026
027import javax.annotation.ParametersAreNonnullByDefault;
028
029import com.google.common.base.Function;
030import com.google.common.base.Functions;
031
032import de.learnlib.algorithms.features.observationtable.writer.ObservationTableHTMLWriter;
033import de.learnlib.algorithms.features.observationtable.writer.ObservationTableWriter;
034
035import net.automatalib.words.Word;
036
037@ParametersAreNonnullByDefault
038public abstract class OTUtils {
039        
040        private static final long BROWSER_STARTUP_DELAY = 5000; // milliseconds
041
042        private static final String HTML_FILE_HEADER = "<html><head>\n"
043                        + "<meta http-equiv=\"content-type\" content=\"text/html; charset=UTF-8\">\n"
044                        + "<style type=\"text/css\">\n"
045                        + "table.learnlib-observationtable { border-width: 1px; border: solid; }\n"
046                        + "table.learnlib-observationtable th.suffixes-header { text-align: center; }\n"
047                        + "table.learnlib-observationtable th.prefix { vertical-align: top; }\n"
048                        + "table.learnlib-observationtable .suffix-column { text-align: left; }\n"
049                        + "table.learnlib-observationtable tr { border-width: 1px; border: solid; }\n"
050                        + "table.learnlib-observationtable tr.long-prefix { background-color: #dfdfdf; }\n"
051                        + "</style></head>\n"
052                        + "<body>\n";
053        private static final String HTML_FILE_FOOTER = "</body></html>\n";
054        
055        
056        public static <I,O>
057        String toString(
058                        ObservationTable<? extends I,? extends O> table,
059                        ObservationTableWriter<I,O> writer) {
060                StringBuilder sb = new StringBuilder();
061                writer.write(table, sb);
062                
063                return sb.toString();
064        }
065        
066        
067        
068        
069        public static <I,O>
070        void writeHTMLToFile(
071                        ObservationTable<I, O> table,
072                        File file,
073                        Function<? super Word<? extends I>,? extends String> wordToString,
074                        Function<? super O,? extends String> outputToString) throws IOException {
075                
076                try(BufferedWriter bw = new BufferedWriter(new FileWriter(file))) {
077                        bw.write(HTML_FILE_HEADER);
078                        ObservationTableHTMLWriter<I, O> otWriter
079                                = new ObservationTableHTMLWriter<>(wordToString, outputToString);
080                        otWriter.write(table, bw);
081                        bw.write(HTML_FILE_FOOTER);
082                }
083        }
084        
085        public static <I,O>
086        void writeHTMLToFile(
087                        ObservationTable<I, O> table,
088                        File file) throws IOException {
089                writeHTMLToFile(table, file, Functions.toStringFunction(), Functions.toStringFunction());
090        }
091        
092        /**
093         * Displays the observation table as a HTML document in the default browser.
094         * <p>
095         * This method internally relies on {@link Desktop#browse(java.net.URI)}, hence it will not
096         * work if {@link Desktop} is not supported, or if the application is running in headless mode.
097         * <p>
098         * <b>IMPORTANT NOTE:</b> Calling this method may delay the termination of the JVM by up to 5 seconds.
099         * This is due to the fact that the temporary file created in this method is marked for deletion
100         * upon JVM termination. If the JVM terminates too early, it might be deleted before it was loaded
101         * by the browser.
102         *   
103         * @param table the observation table to display
104         * @param wordToString the transformation from words to strings. This transformation is <b>not</b> required
105         * nor expected to escape HTML entities
106         * @param outputToString the transformation from outputs to strings. This transformation is <b>not</b> required
107         * nor expected to escape HTML entities
108         * @throws IOException if creating or writing to the temporary file fails 
109         * @throws HeadlessException if the JVM is running in headless mode
110         * @throws UnsupportedOperationException if {@link Desktop#getDesktop()} is not supported by the system
111         */
112        public static <I,O>
113        void displayHTMLInBrowser(
114                        ObservationTable<I,O> table,
115                        Function<? super Word<? extends I>,? extends String> wordToString,
116                        Function<? super O,? extends String> outputToString) throws IOException, HeadlessException, UnsupportedOperationException {
117                File tempFile = File.createTempFile("learnlib-ot" , ".html");
118                tempFile.deleteOnExit();
119                writeHTMLToFile(table, tempFile, wordToString, outputToString);
120                
121                Desktop desktop = Desktop.getDesktop();
122                // We use browse() instead of open() because, e.g., web developers may have
123                // an HTML editor set up as their default application to open HTML files
124                desktop.browse(tempFile.toURI());
125                
126                // Enforce a delayed shutdown of the JVM, in order to make sure
127                // tempFile doesn't get deleted prematurely
128                new FutureTask<>(new Runnable() {
129                        @Override
130                        public void run() {
131                                try {
132                                        Thread.sleep(BROWSER_STARTUP_DELAY);
133                                }
134                                catch(InterruptedException ex) {
135                                        ex.printStackTrace(); // not much we can do here
136                                }
137                        }
138                }, null).run();
139        }
140        
141        public static <I,O>
142        void displayHTMLInBrowser(ObservationTable<I,O> table) throws IOException, HeadlessException, UnsupportedOperationException {
143                displayHTMLInBrowser(table, Functions.toStringFunction(), Functions.toStringFunction());
144        }
145        
146        
147        
148        private OTUtils() {
149                throw new AssertionError("Constructor should never be invoked");
150        }
151}
