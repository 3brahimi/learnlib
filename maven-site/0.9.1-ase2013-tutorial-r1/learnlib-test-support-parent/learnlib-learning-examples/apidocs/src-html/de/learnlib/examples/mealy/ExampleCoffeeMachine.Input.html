018package de.learnlib.examples.mealy;
019
020import static de.learnlib.examples.mealy.ExampleCoffeeMachine.Input.BUTTON;
021import static de.learnlib.examples.mealy.ExampleCoffeeMachine.Input.CLEAN;
022import static de.learnlib.examples.mealy.ExampleCoffeeMachine.Input.POD;
023import static de.learnlib.examples.mealy.ExampleCoffeeMachine.Input.WATER;
024import net.automatalib.automata.transout.MutableMealyMachine;
025import net.automatalib.automata.transout.impl.compact.CompactMealy;
026import net.automatalib.util.automata.builders.AutomatonBuilders;
027import net.automatalib.words.Alphabet;
028import net.automatalib.words.impl.Alphabets;
029import de.learnlib.examples.DefaultLearningExample.DefaultMealyLearningExample;
030import de.learnlib.examples.mealy.ExampleCoffeeMachine.Input;
031
032/**
033 * This example represents the Coffee Machine example from
034 * Steffen et al. "Introduction to Active Automata Learning from
035 * a Practical Perspective" (Figure 3)
036 * 
037 * @author Maik Merten <maikmerten@googlemail.com>
038 */
039public class ExampleCoffeeMachine extends DefaultMealyLearningExample<Input,String> {
040
041
042        public static enum Input {
043                WATER,
044                POD,
045                BUTTON,
046                CLEAN
047        }
048        
049        public final static String out_ok = "ok";
050        public final static String out_error = "error";
051        public final static String out_coffee = "coffee!";
052        
053        
054        public static Alphabet<Input> createInputAlphabet() {
055                return Alphabets.fromEnum(Input.class);
056        }
057        
058        
059        
060    /**
061     * Construct and return a machine representation of this example
062     * 
063     * @return a Mealy machine representing the coffee machine example
064     */   
065    public static <S,T,A extends MutableMealyMachine<S,? super Input,T,? super String>>
066    A constructMachine(A machine) {
067//              S a = machine.addInitialState(), b = machine.addState(), c = machine
068//                              .addState(), d = machine.addState(), e = machine.addState(), f = machine
069//                              .addState();
070//
071//              machine.addTransition(a, WATER, c, out_ok);
072//              machine.addTransition(a, POD, b, out_ok);
073//              machine.addTransition(a, BUTTON, f, out_error);
074//              machine.addTransition(a, CLEAN, a, out_ok);
075//
076//              machine.addTransition(b, WATER, d, out_ok);
077//              machine.addTransition(b, POD, b, out_ok);
078//              machine.addTransition(b, BUTTON, f, out_error);
079//              machine.addTransition(b, CLEAN, a, out_ok);
080//
081//              machine.addTransition(c, WATER, c, out_ok);
082//              machine.addTransition(c, POD, d, out_ok);
083//              machine.addTransition(c, BUTTON, f, out_error);
084//              machine.addTransition(c, CLEAN, a, out_ok);
085//
086//              machine.addTransition(d, WATER, d, out_ok);
087//              machine.addTransition(d, POD, d, out_ok);
088//              machine.addTransition(d, BUTTON, e, out_coffee);
089//              machine.addTransition(d, CLEAN, a, out_ok);
090//
091//              machine.addTransition(e, WATER, f, out_error);
092//              machine.addTransition(e, POD, f, out_error);
093//              machine.addTransition(e, BUTTON, f, out_error);
094//              machine.addTransition(e, CLEAN, a, out_ok);
095//
096//              machine.addTransition(f, WATER, f, out_error);
097//              machine.addTransition(f, POD, f, out_error);
098//              machine.addTransition(f, BUTTON, f, out_error);
099//              machine.addTransition(f, CLEAN, f, out_error);
100
101        machine = AutomatonBuilders.forMealy(machine)
102                        .withInitial("a")
103                        .from("a")
104                                .on(WATER).withOutput(out_ok).to("c")
105                                .on(POD).withOutput(out_ok).to("b")
106                                .on(BUTTON).withOutput(out_error).to("f")
107                                .on(CLEAN).withOutput(out_ok).loop()
108                        .from("b")
109                                .on(WATER).withOutput(out_ok).to("d")
110                                .on(POD).withOutput(out_ok).loop()
111                                .on(BUTTON).withOutput(out_error).to("f")
112                                .on(CLEAN).withOutput(out_ok).to("a")
113                        .from("c")
114                                .on(WATER).withOutput(out_ok).loop()
115                                .on(POD).withOutput(out_ok).to("d")
116                                .on(BUTTON).withOutput(out_error).to("f")
117                                .on(CLEAN).withOutput(out_ok).to("a")
118                        .from("d")
119                                .on(WATER, POD).withOutput(out_ok).loop()
120                                .on(BUTTON).withOutput(out_coffee).to("e")
121                                .on(CLEAN).withOutput(out_ok).to("a")
122                        .from("e")
123                                .on(WATER, POD, BUTTON).withOutput(out_error).to("f")
124                                .on(CLEAN).withOutput(out_ok).to("a")
125                        .from("f")
126                                .on(WATER, POD, BUTTON, CLEAN).withOutput(out_error).loop()
127                .create();
128        
129        
130        return machine;
131    }
132    
133    public static CompactMealy<Input,String> constructMachine() {
134        return constructMachine(new CompactMealy<Input,String>(createInputAlphabet()));
135    }
136
137    
138    public static ExampleCoffeeMachine createExample() {
139        return new ExampleCoffeeMachine();
140    }
141        
142        public ExampleCoffeeMachine() {
143                super(constructMachine());
144        }
145        
146}
