017package de.learnlib.filters.reuse;
018
019import com.google.common.base.Supplier;
020import de.learnlib.api.MembershipOracle.MealyMembershipOracle;
021import de.learnlib.api.Query;
022import de.learnlib.filters.reuse.ReuseCapableOracle.QueryResult;
023import de.learnlib.filters.reuse.tree.ReuseNode;
024import de.learnlib.filters.reuse.tree.ReuseNode.NodeResult;
025import de.learnlib.filters.reuse.tree.ReuseTree;
026import de.learnlib.filters.reuse.tree.ReuseTree.ReuseTreeBuilder;
027import de.learnlib.filters.reuse.tree.SystemStateHandler;
028import net.automatalib.words.Alphabet;
029import net.automatalib.words.Word;
030import net.automatalib.words.WordBuilder;
031
032import java.util.Collection;
033import java.util.Set;
034
035/**
036 * The reuse oracle is a {@link MealyMembershipOracle} that is able to
037 * <ul>
038 * <li>Cache queries: Each processed query will not be delegated again (instead
039 * the answer will be retrieved from the {@link ReuseTree})</li>
040 * <li>Pump queries: If the {@link ReuseTree} is configured to know which
041 * symbols are model invariant input symbols via
042 * {@link ReuseOracleBuilder#withInvariantInputs(Set)} (like a read from a
043 * database which does not change the SUL) or configured for failure output
044 * symbols via {@link ReuseOracleBuilder#withFailureOutputs(Set)} (e.g. a roll
045 * back mechanism exists for the invoked symbol) the oracle could ''pump'' those
046 * symbols inside a query once seen.</li>
047 * <li>Reuse system states: There are a lot of situations where a prefix of a
048 * query is already known and a system state is available. In this situation the
049 * oracle is able to reuse the available system state and only process the
050 * remaining suffix. Whether or not a system state will be removed after it is
051 * used is decided upon construction 
052 * (see {@link ReuseOracleBuilder#ReuseOracleBuilder(Alphabet, Supplier)}.</li>
053 * </ul>
054 * through an internal {@link ReuseTree}.
055 * 
056 * The usage of model invariant input symbols and failure output symbols is
057 * disabled by default and can be enabled upon construction (see {@link ReuseOracleBuilder#withFailureOutputs(Set)} and
058 * {@link ReuseOracleBuilder#withInvariantInputs(Set)}).
059 * 
060 * @author Oliver Bauer <oliver.bauer@tu-dortmund.de>
061 * 
062 * @param <S> system state class
063 * @param <I> input symbol class
064 * @param <O> output symbol class
065 */
066public class ReuseOracle<S, I, O> implements MealyMembershipOracle<I, O> {
067        private final Supplier<? extends ReuseCapableOracle<S, I, O>> oracleSupplier;
068
069        private final ThreadLocal<ReuseCapableOracle<S, I, O>> executableOracles =
070                        new ThreadLocal<ReuseCapableOracle<S, I, O>>() {
071                                @Override
072                                protected ReuseCapableOracle<S, I, O> initialValue() {
073                                        return ReuseOracle.this.oracleSupplier.get();
074                                }
075                        };
076
077        private final ReuseTree<S, I, O> tree;
078
079        public static class ReuseOracleBuilder<S,I,O> {
080                private final Alphabet<I> alphabet;
081                private final Supplier<? extends ReuseCapableOracle<S, I, O>> oracleSupplier;
082
083                private boolean invalidateSystemstates = true;
084                private SystemStateHandler<S> systemStateHandler;
085                private Set<I> invariantInputSymbols;
086                private Set<O> failureOutputSymbols;      
087                
088                public ReuseOracleBuilder(
089                                Alphabet<I> alphabet,
090                                Supplier<? extends ReuseCapableOracle<S, I, O>> oracleSupplier) {
091                        this.alphabet = alphabet;
092                        this.oracleSupplier = oracleSupplier;
093                }
094                
095                public ReuseOracleBuilder<S,I,O> withSystemStateHandler(SystemStateHandler<S> systemStateHandler) {
096                        this.systemStateHandler = systemStateHandler;
097                        return this;
098                }
099
100                public ReuseOracleBuilder<S,I,O> withEnabledSystemstateInvalidation(boolean invalidate) {
101                        this.invalidateSystemstates = invalidate;
102                        return this;
103                }
104                
105                public ReuseOracleBuilder<S,I,O> withInvariantInputs(Set<I> inputs) {
106                        this.invariantInputSymbols = inputs;
107                        return this;
108                }
109                
110                public ReuseOracleBuilder<S,I,O> withFailureOutputs(Set<O> outputs) {
111                        this.failureOutputSymbols = outputs;
112                        return this;
113                }
114                
115                public ReuseOracle<S, I, O> build() {
116                        return new ReuseOracle<>(this);
117                }
118        }
119        
120        /**
121         * Default constructor.
122         */
123        private ReuseOracle(ReuseOracleBuilder<S,I,O> builder) {
124                this.oracleSupplier = builder.oracleSupplier;
125                this.tree = new ReuseTreeBuilder<S,I,O>(builder.alphabet)
126                                .withSystemStateHandler(builder.systemStateHandler)
127                                .withFailureOutputs(builder.failureOutputSymbols)
128                                .withInvariantInputs(builder.invariantInputSymbols)
129                                .withEnabledSystemstateInvalidation(builder.invalidateSystemstates)
130                                .build();
131        }
132
133        /**
134         * {@inheritDoc}.
135         */
136        @Override
137        public void processQueries(Collection<? extends Query<I, Word<O>>> queries) {
138                for (Query<I, Word<O>> query : queries) {
139                        Word<O> output = processQuery(query.getInput());
140                        query.answer(output.suffix(query.getSuffix().size()));
141                }
142        }
143
144        /**
145         * This methods returns the full output to the input query.
146         * <p>
147         * It is possible that the query is already known (answer provided by
148         * {@link ReuseTree#getOutput(Word)}, the query is new and no system state
149         * could be found for reusage ({@link ReuseCapableOracle#processQuery(Word)}
150         * will be invoked) or there exists a prefix that (maybe epsilon) could be
151         * reused so save reset invocation (
152         * {@link ReuseCapableOracle#continueQuery(Word, Object)} will be invoked
153         * with remaining suffix and the corresponding {@link ReuseNode} of the
154         * {@link ReuseTree}).
155         *
156         * @param query
157         * @return
158         */
159        private Word<O> processQuery(final Word<I> query) {
160                Word<O> knownOutput = tree.getOutput(query);
161
162                if (knownOutput != null) {
163                        return knownOutput;
164                }
165
166                NodeResult<S,I,O> nodeResult = tree.fetchSystemState(query);
167
168                ReuseCapableOracle<S, I, O> oracle = getReuseCapableOracle();
169
170                if (nodeResult == null) {
171                        QueryResult<S, O> res = oracle.processQuery(query);
172                        tree.insert(query, res);
173
174                        return res.output;
175                }
176                
177                Word<I> suffix = query.suffix(query.size() - nodeResult.prefixLength);
178                Word<I> prefix = query.prefix(nodeResult.prefixLength);
179                
180                ReuseNode<S, I, O> reuseNode = nodeResult.reuseNode;
181                S systemState = nodeResult.systemState;
182                
183                QueryResult<S, O> queryResult;
184                queryResult = oracle.continueQuery(suffix, systemState);
185                this.tree.insert(suffix, reuseNode, queryResult);
186                
187                Word<O> prefixOutput = tree.getOutput(prefix); // TODO don't compute twice
188                Word<O> suffixOutput = queryResult.output;
189                return new WordBuilder<>(prefixOutput).append(suffixOutput).toWord();
190        }
191
192        /**
193         * Returns the {@link ReuseTree} used by this instance.
194         *
195         * @return
196         */
197        public ReuseTree<S, I, O> getReuseTree() {
198                return this.tree;
199        }
200
201        /**
202         * Returns the {@link ReuseCapableOracle} used by this instance.
203         *
204         * @return
205         */
206        public ReuseCapableOracle<S, I, O> getReuseCapableOracle() {
207                return executableOracles.get();
208        }
