017package de.learnlib.algorithms.features.globalsuffixes;
018
019import java.util.Collection;
020
021import net.automatalib.words.Word;
022
023public interface GlobalSuffixFeature<I> {
024        /**
025         * Retrieves the global suffixes of this learner. Calling this method before {@link #startLearning()}
026         * should return an empty collection.
027         * <p>
028         * The return value should not be modified; attempting to do so may result in an
029         * {@link UnsupportedOperationException}. It is the implementation's responsibility
030         * to ensure attempted modifications do not corrupt the learner's internal state.
031         *  
032         * @return the global suffixes used by this algorithm
033         */
034        public Collection<? extends Word<I>> getGlobalSuffixes();
035        
036        /**
037         * Add the provided suffixes to the collection of global suffixes. As this method 
038         * is designed to possibly trigger a <em>refinement</em>, it is illegal to invoke
039         * it before {@link #startLearning()} has been called.
040         * <p>
041         * The implementation may choose to (but is not required to) omit suffixes which are
042         * already present (that is, manage the global suffixes as a proper set).
043         * 
044         * @param globalSuffixes the global suffixes to add
045         * @return {@code true} if a refinement was triggered by adding the global suffixes,
046         * {@code false otherwise}.
047         */
048        public boolean addGlobalSuffixes(Collection<? extends Word<I>> globalSuffixes);
049}
