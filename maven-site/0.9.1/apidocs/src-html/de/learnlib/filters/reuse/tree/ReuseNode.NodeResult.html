017package de.learnlib.filters.reuse.tree;
018
019import java.util.Arrays;
020import java.util.Collection;
021
022/**
023 * A {@link ReuseNode} is a vertex in the {@link ReuseTree} that contains (a
024 * possible empty) set of outgoing {@link ReuseEdge}s. Each {@link ReuseNode}
025 * may contain a system state holding relevant informations (e.g.
026 * database identifiers or an object) that belongs to the system state that 
027 * 'represents' the system state after executing a membership query.
028 * 
029 * @author Oliver Bauer <oliver.bauer@tu-dortmund.de>
030 * 
031 * @param <S> system state class
032 * @param <I> input symbol class
033 * @param <O> output symbol class
034 */
035public class ReuseNode<S, I, O> {
036        public static final class NodeResult<S,I,O> {
037                public final ReuseNode<S, I, O> reuseNode;
038                public final S systemState;
039                /**
040                 * The prefix length for a membership query that leads to
041                 * the {@link ReuseNode} in the reuse tree.
042                 */
043                public final int prefixLength;
044                public NodeResult(ReuseNode<S, I, O> reuseNode, S systemState, int prefixLength) {
045                        super();
046                        this.reuseNode = reuseNode;
047                        this.systemState = systemState;
048                        this.prefixLength = prefixLength;
049                }
050        }
051        
052        private final ReuseEdge<S,I,O>[] edges;
053        private S systemstate;
054        private final int id;
055
056        @SuppressWarnings("unchecked")
057        public ReuseNode(int id, int alphabetSize) {
058                this.edges = new ReuseEdge[alphabetSize];
059                this.id = id;
060        }
061
062        /**
063         * The system state, may be {@code null}.
064         * 
065         * @return
066         */
067        public S getSystemState() {
068                return systemstate;
069        }
070
071        public void setSystemState(S state) {
072                this.systemstate = state;
073        }
074        /**
075         * Returns all outgoing {@link ReuseEdge}s from this {@link ReuseNode}. If
076         * there are none the returned {@link java.util.Collection} will be empty
077         * (but never {@code null}).
078         * 
079         * @return
080         */
081        public Collection<ReuseEdge<S, I, O>> getEdges() {
082                return Arrays.asList(edges);
083        }
084
085        /**
086         * Adds an outgoing {@link ReuseEdge} to this {@link ReuseNode}.
087         * 
088         * @param edge
089         */
090        public void addEdge(int index, ReuseEdge<S, I, O> edge) {
091                this.edges[index] = edge;
092        }
093
094        /**
095         * May be {@code null}.
096         *
097         * @param index
098         * @return
099         */
100        public ReuseEdge<S, I, O> getEdgeWithInput(int index) {
101                return this.edges[index];
102        }
103
104        public ReuseNode<S, I, O> getTargetNodeForInput(int index) {
105                ReuseEdge<S, I, O> edge = this.getEdgeWithInput(index);
106                if (edge == null) {
107                        return null;
108                }
109                return edge.getTarget();
110        }
111
112        public int getId() {
113                return this.id;
114        }
115        
116        @Override
117        public int hashCode() {
118                final int prime = 31;
119                int result = 1;
120                result = prime * result + id;
121                return result;
122        }
123
124        @Override
125        public boolean equals(Object obj) {
126                if (obj instanceof ReuseNode) {
127                        @SuppressWarnings("unchecked")
128                        ReuseNode<S, I, O> other = ((ReuseNode<S,I,O>)obj);
129                        return other.id == id;
130                }
131                return false;
132        }
133}
