017package de.learnlib.algorithms.features.observationtable.writer;
018
019import java.io.IOException;
020import java.util.List;
021
022import net.automatalib.words.Word;
023
024import com.google.common.base.Function;
025
026import de.learnlib.algorithms.features.observationtable.ObservationTable;
027import de.learnlib.algorithms.features.observationtable.ObservationTable.Row;
028
029/**
030 * Writes an observation table as a HTML table.
031 * <p>
032 * 
033 * @author Malte Isberner <malte.isberner@gmail.com>
034 *
035 * @param <I> input symbol type (upper bound)
036 * @param <O> output symbol type (upper bound)
037 */
038public class ObservationTableHTMLWriter<I, O> extends AbstractObservationTableWriter<I,O> {
039        
040        public ObservationTableHTMLWriter(
041                        Function<? super Word<? extends I>,? extends String> wordToString,
042                        Function<? super O,? extends String> outputToString) {
043                super(wordToString, outputToString);
044        }
045        
046        @Override
047        public void write(ObservationTable<? extends I,? extends O> table, Appendable out) throws IOException {
048                List<? extends Word<? extends I>> suffixes = table.getSuffixes();
049                
050                out.append("<table class=\"learnlib-observationtable\">\n");
051                out.append("    <thead>\n");
052                out.append("            <tr><th rowspan=\"2\" class=\"prefix\">Prefix</th><th colspan=\"").append(Integer.toString(suffixes.size())).append("\" class=\"suffixes-header\">Suffixes</th></tr>\n");
053                out.append("            <tr>");
054                for(Word<? extends I> suffix : suffixes) {
055                        out.append("<td>").append(wordToString(suffix)).append("</td>");
056                }
057                out.append("</tr>\n");
058                out.append("    </thead>\n");
059                out.append("    <tbody>\n");
060                
061                for(Row<? extends I,? extends O> row : table.getShortPrefixRows()) {
062                        out.append("            <tr class=\"short-prefix\"><td class=\"prefix\">").append(wordToString(row.getLabel())).append("</td>");
063                        for(O value : row) {
064                                out.append("<td class=\"suffix-column\">").append(outputToString(value)).append("</td>");
065                        }
066                        out.append("</tr>\n");
067                }
068                
069                out.append("            <tr><td colspan=\"").append(Integer.toString(suffixes.size() + 1)).append("\"></td></tr>\n");
070                
071                for(Row<? extends I,? extends O> row : table.getLongPrefixRows()) {
072                        out.append("            <tr class=\"long-prefix\"><td>").append(wordToString(row.getLabel())).append("</td>");
073                        for(O value : row) {
074                                out.append("<td class=\"suffix-column\">").append(outputToString(value)).append("</td>");
075                        }
076                        out.append("</tr>\n");
077                }
078                
079                out.append("</table>\n");
080        }
081        
082        
083}
