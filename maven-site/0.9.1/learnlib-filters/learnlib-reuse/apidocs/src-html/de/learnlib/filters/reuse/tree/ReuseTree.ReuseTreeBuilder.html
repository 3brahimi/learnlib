039/**
040 * The {@link ReuseTree} is a tree like structure consisting of nodes (see
041 * {@link ReuseNode}) and edges (see {@link ReuseEdge}) that is used by the
042 * {@link ReuseOracle}:
043 * <ul>
044 * <li>Nodes may contain a system state (see {@link ReuseNode#getSystemState()})
045 * that could be used for executing suffixes of membership queries. Each node
046 * consists of a (possible empty) set of outgoing edges.
047 * <li>Edges consists beside source and target node of input and output
048 * behavior.
049 * </ul>
050 * The {@link ReuseTree} is the central data structure that maintains observed
051 * behavior from the SUL and maintains also available system states. The
052 * {@link ReuseTree} is only 'tree like' since it may contain reflexive edges at
053 * nodes (only possible if {@link ReuseTreeBuilder#withFailureOutputs(Set)} or
054 * {@link ReuseTreeBuilder#withInvariantInputs(Set)} is set).
055 * 
056 * @author Oliver Bauer <oliver.bauer@tu-dortmund.de>
057 * 
058 * @param <S> system state class
059 * @param <I> input symbol class
060 * @param <O> output symbol class
061 */
062public class ReuseTree<S, I, O> extends AbstractGraph<ReuseNode<S, I, O>, ReuseEdge<S, I, O>>
063        implements DOTPlottableGraph<ReuseNode<S, I, O>, ReuseEdge<S, I, O>> {
064        
065        
066        public static class ReuseTreeBuilder<S,I,O> {
067                // mandatory
068                private final Alphabet<I> alphabet;
069                
070                // optional
071                private boolean invalidateSystemstates = true;
072                private SystemStateHandler<S> systemStateHandler;
073                private Set<I> invariantInputSymbols;
074                private Set<O> failureOutputSymbols;              
075                
076                public ReuseTreeBuilder(Alphabet<I> alphabet) {
077                        this.alphabet = alphabet;
078                        this.systemStateHandler = null;
079                        this.invariantInputSymbols = Collections.emptySet();
080                        this.failureOutputSymbols = Collections.emptySet();
081                }
082                
083                public ReuseTreeBuilder<S,I,O> withSystemStateHandler(SystemStateHandler<S> systemStateHandler) {
084                        this.systemStateHandler = systemStateHandler;
085                        return this;
086                }
087
088                public ReuseTreeBuilder<S,I,O> withEnabledSystemstateInvalidation(boolean invalidate) {
089                        this.invalidateSystemstates = invalidate;
090                        return this;
091                }
092                
093                public ReuseTreeBuilder<S,I,O> withInvariantInputs(Set<I> inputs) {
094                        this.invariantInputSymbols = inputs;
095                        return this;
096                }
097                
098                public ReuseTreeBuilder<S,I,O> withFailureOutputs(Set<O> outputs) {
099                        this.failureOutputSymbols = outputs;
100                        return this;
101                }
102                
103                public ReuseTree<S, I, O> build() {
104                        return new ReuseTree<>(this);
105                }
106        }
107        
108        private final Alphabet<I> alphabet;
109        private final int alphabetSize;
110        
111        private final Set<I> invariantInputSymbols;
112        private final Set<O> failureOutputSymbols;
113
114        private final boolean invalidateSystemstates;
115        private final SystemStateHandler<S> systemStateHandler;
116
117        /** Maybe reset to zero, see {@link ReuseTree#clearTree()} */
118        private int nodeCount = 0;
119        /** Maybe reinitialized , see {@link ReuseTree#clearTree()} */
120        private ReuseNode<S, I, O> root;
121        
122        private ReuseTree(ReuseTreeBuilder<S,I,O> builder) {
123                this.alphabet = builder.alphabet;
124                this.invalidateSystemstates = builder.invalidateSystemstates;
125                SystemStateHandler<S> handler = builder.systemStateHandler;
126                // If the specified handler is null, no action is required
127                if(handler == null) {
128                        handler = new SystemStateHandler<S>() {
129                                @Override
130                                public void dispose(S state) {
131                                }
132                        };
133                }
134                this.systemStateHandler = handler;
135                this.invariantInputSymbols = 
136                                (builder.invariantInputSymbols != null) ? builder.invariantInputSymbols : Collections.<I>emptySet();
137                this.failureOutputSymbols =
138                                (builder.failureOutputSymbols != null) ? builder.failureOutputSymbols : Collections.<O>emptySet();
139                
140                // local and not configurable
141                this.alphabetSize = alphabet.size();
142                this.root = new ReuseNode<>(nodeCount++, alphabetSize);
143        }
144
145        /**
146         * Returns the root {@link ReuseNode} of the {@link ReuseTree}.
147         *
148         * @return root The root of the tree, never {@code null}.
149         */
150        public final ReuseNode<S, I, O> getRoot() {
151                return this.root;
152        }
153
154        /**
155         * Returns the known output for the given query or {@code null} if not
156         * known.
157         * 
158         * @param query
159         *            Not allowed to be {@code null}.
160         * @return The output for {@code query} if already known from the
161         *         {@link ReuseTree} or {@code null} if unknown.
162         */
163        public final synchronized Word<O> getOutput(final Word<I> query) {
164                if (query == null) {
165                        String msg = "Query is not allowed to be null.";
166                        throw new IllegalArgumentException(msg);
167                }
168
169                WordBuilder<O> output = new WordBuilder<>();
170
171                ReuseNode<S, I, O> sink = getRoot();
172                ReuseNode<S, I, O> node;
173                ReuseEdge<S, I, O> edge;
174                for (I symbol : query) {
175                        int index = alphabet.getSymbolIndex(symbol);
176                        edge = sink.getEdgeWithInput(index);
177
178                        if (edge == null) {
179                                return null;
180                        }
181
182                        node = edge.getTarget();
183                        output.add(edge.getOutput());
184                        sink = node;
185                }
186
187                return output.toWord();
188        }
189
190        /**
191         * This method removes all system states from the tree. The tree structure
192         * remains, but there will be nothing for reusage.
193         * <p>
194         * The {@link SystemStateHandler} (
195         * {@link #setSystemStateHandler(SystemStateHandler)}) will be informed
196         * about all disposings.
197         */
198        public final synchronized void disposeSystemstates() {
199                disposeSystemstates(getRoot());
200        }
201
202        private void disposeSystemstates(ReuseNode<S, I, O> node) {
203                if (node.getSystemState() != null) {
204                        systemStateHandler.dispose(node.getSystemState());
205                }
206                node.setSystemState(null);
207                for (ReuseEdge<S, I, O> edge : node.getEdges()) {
208                        if (edge != null) {
209                                if (!edge.getTarget().equals(node)) {
210                                        // only for non reflexive edges, there are no circles in a tree
211                                        disposeSystemstates(edge.getTarget());
212                                }
213                        }
214                }
215        }
216
217        /**
218         * Clears the whole tree which means the root will be reinitialized by a new
219         * {@link ReuseNode} and all invariant input symbols as well as all failure
220         * output symbols will be removed.
221         * <p>
222         * The {@link SystemStateHandler} (
223         * {@link #setSystemStateHandler(SystemStateHandler)}) will <b>not</b> be
224         * informed about any disposings.
225         */
226        public synchronized void clearTree() {
227                this.nodeCount = 0;
228                this.root = new ReuseNode<>(nodeCount++, alphabetSize);
229                this.invariantInputSymbols.clear();
230                this.failureOutputSymbols.clear();
231        }
232
233        /**
234         * Returns a reuseable {@link NodeResult} with system state 
235         * or {@code null} if none such exists. If
236         * ''oldInvalidated'' was set to {@code true} (in the {@link ReuseOracle})
237         * the system state is already removed from the tree whenever
238         * one was available.
239         * 
240         * @param query
241         *            Not allowed to be {@code null}.
242         * @return
243         */
244        public synchronized NodeResult<S,I,O> fetchSystemState(Word<I> query) {
245                if (query == null) {
246                        String msg = "Query is not allowed to be null.";
247                        throw new IllegalArgumentException(msg);
248                }
249
250                int length = 0;
251
252                ReuseNode<S, I, O> sink = getRoot();
253                ReuseNode<S, I, O> lastState = null;
254                if (sink.getSystemState() != null) {
255                        lastState = sink;
256                }
257
258                ReuseNode<S, I, O> node;
259                for (int i = 0; i < query.size(); i++) {
260                        node = sink.getTargetNodeForInput(alphabet.getSymbolIndex(query.getSymbol(i)));
261
262                        if (node == null) {
263                                // we have reached longest known prefix
264                                break;
265                        }
266
267                        sink = node;
268                        if (sink.getSystemState() != null) {
269                                lastState = sink;
270                                length = i + 1;
271                        }
272                }
273
274                if (lastState == null) {
275                        return null;
276                }
277                S systemState = lastState.getSystemState();
278                if (invalidateSystemstates) {
279                        lastState.setSystemState(null);
280                }
281                return new NodeResult<>(lastState, systemState, length);
282        }

284        /**
285         * Inserts the given {@link Word} with {@link QueryResult} into the tree
286         * starting from the root node of the tree. For the longest known prefix of
287         * the given {@link Word} there will be no new nodes or edges created.
288         * <p>
289         * Will be called from the {@link ReuseOracle} if no system state was
290         * available for reusage for the query (otherwise
291         * {@link #insert(Word, ReuseNode, QueryResult)} would be called). The last
292         * node reached by the last symbol of the query will hold the system state
293         * from the given {@link QueryResult}.
294         * <p>
295         * This method should only be invoked internally from the
296         * {@link ReuseOracle} unless you know exactly what you are doing (you may
297         * want to create a predefined reuse tree before start learning).
298         * 
299         * @param query
300         * @param queryResult
301         * 
302         * @throws ReuseException if non deterministic behavior is detected
303         */
304        public synchronized void insert(Word<I> query, QueryResult<S, O> queryResult) {
305                insert(query, getRoot(), queryResult);
306        }

308        /**
309         * Inserts the given {@link Word} (suffix of a membership query) with
310         * {@link QueryResult} (suffix output) into the tree starting from the
311         * {@link ReuseNode} (contains prefix with prefix output) in the tree. For
312         * the longest known prefix of the suffix from the given {@link Word} there
313         * will be no new nodes or edges created.
314         * <p>
315         * Will be called from the {@link ReuseOracle} if an available system state
316         * was reused for the query (otherwise {@link #insert(Word, QueryResult)}
317         * would be called). The old system state was already removed from the
318         * {@link ReuseNode} (through {@link #fetchSystemState(Word)}) 
319         * if the ''invalidateSystemstates'' flag in the {@link ReuseOracle}
320         * was set to {@code true}.
321         * <p>
322         * This method should only be invoked internally from the
323         * {@link ReuseOracle} unless you know exactly what you are doing (you may
324         * want to create a predefined reuse tree before start learning).
325         * 
326         * @param query
327         * @param sink
328         * @param queryResult
329         * 
330         * @throws ReuseException if non deterministic behavior is detected
331         */
332        public synchronized void insert(Word<I> query, ReuseNode<S, I, O> sink,
333                        QueryResult<S, O> queryResult) {
334                if (queryResult == null) {
335                        String msg = "The queryResult is not allowed to be null.";
336                        throw new IllegalArgumentException(msg);
337                }
338                if (sink == null) {
339                        String msg = "Node is not allowed to be null, called wrong method?";
340                        throw new IllegalArgumentException(msg);
341                }
342                if (query.size() != queryResult.output.size()) {
343                        String msg = "Size mismatch: " + query + "/" + queryResult.output;
344                        throw new IllegalArgumentException(msg);
345                }
346
347                for (int i = 0; i < query.size(); i++) {
348                        I in = query.getSymbol(i);
349                        O out = queryResult.output.getSymbol(i);
350                        ReuseNode<S, I, O> rn;
351
352                        ReuseEdge<S, I, O> edge = sink.getEdgeWithInput(alphabet.getSymbolIndex(in));
353                        if (edge != null) {
354                                if (Objects.equals(edge.getOutput(), out)) {
355                                        sink = edge.getTarget();
356                                        continue;
357                                }
358
359                                throw new ReuseException(
360                                                "Conflict: input '" + query + "', output '" + queryResult.output + "', i=" + i +
361                                                                ", cached output '" + edge.getOutput() + "'");
362                        }
363
364                        if (failureOutputSymbols.contains(out)) {
365                                rn = sink;
366                        } else if (invariantInputSymbols.contains(in)) {
367                                rn = sink;
368                        } else {
369                                rn = new ReuseNode<>(nodeCount++, alphabetSize);
370                        }
371
372                        int index = alphabet.getSymbolIndex(in);
373                        sink.addEdge(index, new ReuseEdge<>(sink, rn, in, out));
374                        sink = rn;
375                }
376                
377                // If there already is a state stored at this node, overwrite it (but dispose it beforehand)
378                // TODO: Allow storing multiple states at a node
379                S oldState = sink.getSystemState();
380                if(oldState != null) {
381                        systemStateHandler.dispose(oldState);
382                }
383                sink.setSystemState(queryResult.newState);
384        }

386        /*
387         * (non-Javadoc)
388         * @see net.automatalib.graphs.Graph#getNodes()
389         */
390        @Override
391        public Collection<ReuseNode<S, I, O>> getNodes() {
392                Collection<ReuseNode<S, I, O>> collection = new ArrayList<>();
393                appendNodesRecursively(collection, getRoot());
394                return collection;
395        }
396        
397        private void appendNodesRecursively(Collection<ReuseNode<S, I, O>> nodes, ReuseNode<S, I, O> current) {
398                nodes.add(current);
399                for (int i=0; i<alphabetSize; i++) {
400                        ReuseEdge<S, I, O> reuseEdge = current.getEdgeWithInput(i);
401                        if (reuseEdge == null) {
402                                continue;
403                        }
404                        if (!current.equals(reuseEdge.getTarget())) {
405                                appendNodesRecursively(nodes, reuseEdge.getTarget());
406                        }
407                }
408        }
