018package de.learnlib.examples.dfa;
019
020import net.automatalib.automata.fsa.MutableDFA;
021import net.automatalib.automata.fsa.impl.compact.CompactDFA;
022import net.automatalib.util.automata.builders.AutomatonBuilders;
023import net.automatalib.words.Alphabet;
024import net.automatalib.words.impl.Alphabets;
025import de.learnlib.examples.DefaultLearningExample.DefaultDFALearningExample;
026
027/**
028 * This class provides the example used in the paper ''Learning Regular Sets
029 * from Queries and Counterexamples'' by Dana Angluin that consists of an
030 * automaton that accepts ''all strings over {0,1} with an even number of 0's
031 * and an even number of 1's.''
032 * 
033 * @author Oliver Bauer <oliver.bauer@tu-dortmund.de>
034 */
035public class ExampleAngluin extends DefaultDFALearningExample<Integer> {
036
037
038        public static Alphabet<Integer> createInputAlphabet() {
039                return Alphabets.integers(0, 1);
040        }
041        
042        
043        
044        public static <A extends MutableDFA<S, ? super Integer>,S>
045        A constructMachine(A machine) {
046                
047//              S q0 = machine.addInitialState(true);
048//              S q1 = machine.addState(false), q2 = machine.addState(false), q3 = machine
049//                              .addState(false);
050//
051//              machine.addTransition(q0, 0, q1);
052//              machine.addTransition(q0, 1, q2);
053//
054//              machine.addTransition(q1, 0, q0);
055//              machine.addTransition(q1, 1, q3);
056//
057//              machine.addTransition(q2, 0, q3);
058//              machine.addTransition(q2, 1, q0);
059//
060//              machine.addTransition(q3, 0, q2);
061//              machine.addTransition(q3, 1, q1);
062
063        machine = AutomatonBuilders.forDFA(machine)
064                                .from("q0")
065                                        .on(0).to("q1")
066                                        .on(1).to("q2")
067                                .from("q1")
068                                        .on(0).to("q0")
069                                        .on(1).to("q3")
070                                .from("q2")
071                                        .on(0).to("q3")
072                                        .on(1).to("q0")
073                                .from("q3")
074                                        .on(0).to("q2")
075                                        .on(1).to("q3")
076                                .withAccepting("q0")
077                                .withInitial("q0")
078                        .create();
079                
080                return machine;
081        }
082        
083        public static CompactDFA<Integer> constructMachine() {
084                return constructMachine(new CompactDFA<>(createInputAlphabet()));
085        }
086
087        
088        public static ExampleAngluin createExample() {
089                return new ExampleAngluin();
090        }
091
092        
093        
094        public ExampleAngluin() {
095                super(constructMachine());
096        }
097}
