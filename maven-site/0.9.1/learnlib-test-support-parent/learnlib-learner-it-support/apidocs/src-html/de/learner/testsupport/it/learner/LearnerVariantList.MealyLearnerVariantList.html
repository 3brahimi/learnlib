017package de.learner.testsupport.it.learner;
018
019import net.automatalib.automata.fsa.DFA;
020import net.automatalib.automata.transout.MealyMachine;
021import net.automatalib.words.Word;
022import de.learnlib.api.LearningAlgorithm;
023
024/**
025 * A write-only list to store multiple variants of a learning algorithm.
026 * <p>
027 * Usually, there should be one integration test class per learning algorithm. However,
028 * in many cases a single learning algorithm can be configured in numerous ways, all (or many)
029 * of which should be tested independently. Due to the large number of possible combinations,
030 * it is undesirable to create a single integration test class for each configuration; instead,
031 * these variants should be configured and created programmatically. The purpose of the variant
032 * list is to offer a convenient interface for storing all these variants.
033 * 
034 * @author Malte Isberner
035 *
036 * @param <M> hypothesis model type (upper bound)
037 * @param <I> input symbol type
038 * @param <O> output type
039 */
040public interface LearnerVariantList<M, I, O> {
041        
042        public static interface DFALearnerVariantList<I> extends LearnerVariantList<DFA<?,I>,I,Boolean> {}
043        public static interface MealyLearnerVariantList<I,O> extends LearnerVariantList<MealyMachine<?,I,?,O>,I,Word<O>> {}
044        public static interface MealySymLearnerVariantList<I,O> extends LearnerVariantList<MealyMachine<?,I,?,O>,I,O> {}
045        
046        /**
047         * Adds a learner variant with the default maximum number of rounds (i.e., the size of the
048         * target automaton) to the list.
049         * <p>
050         * This is a convenience method, equivalent to invoking {@code addLearnerVariant(name, learner, -1)}.
051         * 
052         * @param name the name of the variant
053         * @param learner the algorithm instance for this variant
054         */
055        public void addLearnerVariant(String name, LearningAlgorithm<? extends M,I,O> learner);
056        
057        /**
058         * Adds a learner variant with a given maximum number of rounds to the list.
059         * 
060         * @param name the name of the variant
061         * @param learner the algorithm instance for this variant
062         * @param maxRounds the maximum number of rounds for the specified target automaton. If a value
063         * less than or equal to zero is specified, the default maximum number of rounds (the size of the
064         * target automaton) is assumed.
065         */
066        public void addLearnerVariant(String name, LearningAlgorithm<? extends M,I,O> learner, int maxRounds);
067
068        
069}
