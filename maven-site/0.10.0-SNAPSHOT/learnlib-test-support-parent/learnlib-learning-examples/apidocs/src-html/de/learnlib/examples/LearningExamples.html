017package de.learnlib.examples;
018
019import java.util.Arrays;
020import java.util.List;
021import java.util.Random;
022
023import net.automatalib.words.Alphabet;
024import net.automatalib.words.impl.Alphabets;
025import de.learnlib.examples.LearningExample.DFALearningExample;
026import de.learnlib.examples.LearningExample.MealyLearningExample;
027import de.learnlib.examples.dfa.ExampleAngluin;
028import de.learnlib.examples.dfa.ExamplePaulAndMary;
029import de.learnlib.examples.mealy.ExampleCoffeeMachine;
030import de.learnlib.examples.mealy.ExampleGrid;
031import de.learnlib.examples.mealy.ExampleRandomMealy;
032import de.learnlib.examples.mealy.ExampleShahbazGroz;
033import de.learnlib.examples.mealy.ExampleStack;
034
035public abstract class LearningExamples {
036        
037        private static long RANDOM_SEED = 1337L;
038        private static final Alphabet<Character> RANDOM_ALPHABET = Alphabets.characters('a', 'c');
039        
040        private static final int RANDOM_SIZE = 100;
041
042        private static final int GRID_XSIZE = 5;
043        private static final int GRID_YSIZE = 5;
044        
045        private static final String[] RANDOM_MEALY_OUTPUTS = { "o1", "o2", "o3" };
046        
047
048        private LearningExamples() {
049                throw new AssertionError("Constructor should not be invoked");
050        }
051        
052        public static List<? extends DFALearningExample<?>> createDFAExamples() {
053                return Arrays.asList(
054                                ExampleAngluin.createExample(),
055                                ExamplePaulAndMary.createExample()
056                                );
057        }
058        
059        public static List<? extends MealyLearningExample<?,?>> createMealyExamples() {
060                return Arrays.asList(
061                                ExampleCoffeeMachine.createExample(),
062                                ExampleGrid.createExample(GRID_XSIZE, GRID_YSIZE),
063                                ExampleShahbazGroz.createExample(),
064                                ExampleStack.createExample(),
065                                ExampleRandomMealy.createExample(new Random(RANDOM_SEED), RANDOM_ALPHABET, RANDOM_SIZE, RANDOM_MEALY_OUTPUTS)
066                                );
067        }
068
069}
