017package de.learnlib.algorithms.lstargeneric.mealy;
018
019import java.util.ArrayList;
020import java.util.Collection;
021import java.util.List;
022
023import net.automatalib.words.Alphabet;
024import net.automatalib.words.Word;
025
026public class LStarMealyUtil {
027        
028        public static <I> List<Word<I>> ensureSuffixCompliancy(Collection<? extends Word<I>> suffixes, Alphabet<I> alphabet,
029                        boolean needsConsistencyCheck) {
030                List<Word<I>> compSuffixes = new ArrayList<Word<I>>();
031                if(needsConsistencyCheck) {
032                        for(int i = 0; i < alphabet.size(); i++)
033                                compSuffixes.add(Word.fromLetter(alphabet.getSymbol(i)));
034                }
035                
036                for(Word<I> w : suffixes) {
037                        if(w.isEmpty())
038                                continue;
039                        if(needsConsistencyCheck && w.length() == 1)
040                                continue;
041                        compSuffixes.add(w);
042                }
043                
044                return compSuffixes;
045        }
046
047}
