017package de.learnlib.algorithms.baselinelstar;

019import java.util.ArrayList;
020import java.util.Collection;
021import java.util.Collections;
022import java.util.LinkedList;
023import java.util.List;
024
025import javax.annotation.Nonnull;
026import javax.annotation.Nullable;
027
028import com.google.common.collect.Lists;
029
030import de.learnlib.algorithms.features.observationtable.AbstractObservationTable;
031
032import net.automatalib.words.Alphabet;
033import net.automatalib.words.Word;

035/**
036 * The internal storage mechanism for {@link BaselineLStar}.
037 *
038 * @param <I>
039 *              input symbol class.
040 */
041public class ObservationTable<I> extends AbstractObservationTable<I, Boolean> {
042
043        @Nonnull
044        private final List<ObservationTableRow<I>> shortPrefixRows; // S
045
046        @Nonnull
047        private final List<ObservationTableRow<I>> longPrefixRows;  // SA
048
049        @Nonnull
050        private final List<Word<I>> suffixes;                          // E
051
052
053        public ObservationTable() {
054                Word<I> emptyWord = Word.epsilon();
055
056                suffixes = new ArrayList<>();
057                suffixes.add(emptyWord);
058
059                Word<I> epsiplon = Word.epsilon();
060                ObservationTableRow<I> initialRow = new ObservationTableRow<>(epsiplon);
061                initialRow.setShortPrefixRow();
062                shortPrefixRows = new LinkedList<>();
063                shortPrefixRows.add(initialRow);
064
065                longPrefixRows = new LinkedList<>();
066        }
067
068        /**
069         * The set of suffixes in the observation table, often called "E".
070         *
071         * @return The set of candidates.
072         */
073        @Override
074        public List<Word<I>> getSuffixes() {
075                return Collections.unmodifiableList(suffixes);
076        }
077
078        void addSuffix(Word<I> suffix) {
079                suffixes.add(suffix);
080        }
081
082        @Override
083        public Collection<ObservationTableRow<I>> getShortPrefixRows() {
084                return Collections.unmodifiableCollection(shortPrefixRows);
085        }
086
087        @Nonnull
088        public List<Word<I>> getShortPrefixLabels() {
089                List<Word<I>> labels = Lists.newArrayListWithExpectedSize(shortPrefixRows.size());
090                for (ObservationTableRow<I> row : shortPrefixRows) {
091                        labels.add(row.getLabel());
092                }
093                return labels;
094        }
095
096        @Override
097        public Collection<ObservationTableRow<I>> getLongPrefixRows() {
098                return Collections.unmodifiableCollection(longPrefixRows);
099        }
100
101        @Nonnull
102        public List<Word<I>> getLongPrefixLabels() {
103                List<Word<I>> labels = Lists.newArrayListWithExpectedSize(longPrefixRows.size());
104                for (ObservationTableRow<I> row : longPrefixRows) {
105                        labels.add(row.getLabel());
106                }
107                return labels;
108        }
109
110        @Override
111        @Nullable
112        public Row<I, Boolean> getSuccessorRow(@Nonnull Row<I, Boolean> spRow, I symbol) {
113                //noinspection SuspiciousMethodCalls
114                if (!shortPrefixRows.contains(spRow)) {
115                        throw new IllegalArgumentException("Row '" + spRow + "' is not part of short prefix rows!");
116                }
117
118                Word<I> successorLabel = spRow.getLabel().append(symbol);
119
120                Row<I, Boolean> successor = null;
121
122                for (Row<I, Boolean> row : getAllRows()) {
123                        if (row.getLabel().equals(successorLabel)) {
124                                successor = row;
125                                break;
126                        }
127                }
128
129                return successor;
130        }
131
132        void addShortPrefix(@Nonnull Word<I> shortPrefix) {
133                final ObservationTableRow<I> row = new ObservationTableRow<>(shortPrefix);
134                row.setShortPrefixRow();
135                shortPrefixRows.add(row);
136        }
137
138        void addLongPrefix(@Nonnull Word<I> longPrefix) {
139                final ObservationTableRow<I> row = new ObservationTableRow<>(longPrefix);
140                row.setLongPrefixRow();
141                longPrefixRows.add(row);
142        }
143
144        void removeShortPrefixesFromLongPrefixes() {
145                List<Word<I>> longPrefixLabels = getLongPrefixLabels();
146                longPrefixLabels.retainAll(getShortPrefixLabels());
147
148                List<ObservationTableRow<I>> rowsToRemove = Lists.newArrayListWithCapacity(longPrefixLabels.size());
149
150                for (ObservationTableRow<I> row : longPrefixRows) {
151                        if (longPrefixLabels.contains(row.getLabel())) {
152                                rowsToRemove.add(row);
153                        }
154                }
155
156                longPrefixRows.removeAll(rowsToRemove);
157        }
158
159        /**
160         * Adds the result of a membership query to this table.
161         *
162         * @param prefix
163         *              The prefix of the {@link Word} asked with the membership query.
164         * @param suffix
165         *              The prefix of the {@link Word} asked with the membership query.
166         * @param result
167         *              The result of the query.
168         */
169        void addResult(@Nonnull Word<I> prefix, @Nonnull Word<I> suffix, @Nonnull Boolean result) {
170                if (!suffixes.contains(suffix)) {
171                        throw new IllegalArgumentException("Suffix '" + suffix + "' is not part of the suffixes set");
172                }
173
174                final int suffixPosition = suffixes.indexOf(suffix);
175                ObservationTableRow<I> row = getRowForPrefix(prefix);
176
177                addResultToRow(result, suffixPosition, row);
178        }
179
180        private void addResultToRow(@Nonnull Boolean result, int suffixPosition, @Nonnull ObservationTableRow<I> row) {
181                final List<Boolean> values = row.getContents();
182                if (values.size() > suffixPosition) {
183                        if (!values.get(suffixPosition).equals(result)) {
184                                throw new IllegalStateException(
185                                                "New result " + values.get(suffixPosition) + " differs from old result " + result);
186                        }
187                }
188                else {
189                        row.addValue(result);
190                }
191        }
192
193        /**
194         * Determines the next state for which the observation table needs to be closed.
195         *
196         * @return The next state for which the observation table needs to be closed. If the
197         * table is closed, this returns {@code null}.
198         */
199        @Nullable
200        Word<I> findUnclosedState() {
201                for (ObservationTableRow<I> candidate : longPrefixRows) {
202                        boolean found = false;
203
204                        for (ObservationTableRow<I> stateRow : shortPrefixRows) {
205                                if (candidate.getContents().equals(stateRow.getContents())) {
206                                        found = true;
207                                        break;
208                                }
209                        }
210
211                        if (!found) {
212                                return candidate.getLabel();
213                        }
214                }
215
216                return null;
217        }
218
219        /**
220         * @param alphabet
221         *              The {@link Alphabet} for which the consistency is checked
222         * @return if the observation table is consistent with the given alphabet.
223         */
224        boolean isConsistentWithAlphabet(@Nonnull Alphabet<I> alphabet) {
225                return findInconsistentSymbol(alphabet) == null;
226        }
227
228        @Nullable
229        InconsistencyDataHolder<I> findInconsistentSymbol(@Nonnull Alphabet<I> alphabet) {
230                for (I symbol : alphabet) {
231                        for (int firstStateCounter = 0; firstStateCounter < shortPrefixRows.size(); firstStateCounter++) {
232                                ObservationTableRow<I> firstRow = shortPrefixRows.get(firstStateCounter);
233
234                                for (int secondStateCounter = firstStateCounter + 1; secondStateCounter < shortPrefixRows.size();
235                                     secondStateCounter++) {
236                                        ObservationTableRow<I> secondRow = shortPrefixRows.get(secondStateCounter);
237
238                                        if (checkInconsistency(firstRow, secondRow, symbol)) {
239                                                return new InconsistencyDataHolder<>(firstRow, secondRow, symbol);
240                                        }
241                                }
242                        }
243                }
244
245                return null;
246        }
247
248        private boolean checkInconsistency(@Nonnull ObservationTableRow<I> firstRow, @Nonnull ObservationTableRow<I> secondRow,
249                        @Nonnull I alphabetSymbol) {
250                ObservationTableRow<I> rowForFirstState = firstRow;
251                ObservationTableRow<I> rowForSecondState = secondRow;
252
253                if (!rowForFirstState.getContents().equals(rowForSecondState.getContents())) {
254                        return false;
255                }
256
257                Word<I> extendedFirstState = firstRow.getLabel().append(alphabetSymbol);
258                Word<I> extendedSecondState = secondRow.getLabel().append(alphabetSymbol);
259                ObservationTableRow<I> rowForExtendedFirstState = getRowForPrefix(extendedFirstState);
260                ObservationTableRow<I> rowForExtendedSecondState = getRowForPrefix(extendedSecondState);
261
262                return !rowForExtendedFirstState.getContents().equals(rowForExtendedSecondState.getContents());
263        }
264
265        @Nonnull
266        Word<I> determineWitnessForInconsistency(@Nonnull InconsistencyDataHolder<I> dataHolder) {
267                Word<I> firstState = dataHolder.getFirstState().append(dataHolder.getDifferingSymbol());
268                Word<I> secondState = dataHolder.getSecondState().append(dataHolder.getDifferingSymbol());
269
270                ObservationTableRow<I> firstRow = getRowForPrefix(firstState);
271                ObservationTableRow<I> secondRow = getRowForPrefix(secondState);
272
273                for (int i = 0; i < firstRow.getContents().size(); i++) {
274                        Boolean symbolFirstRow = firstRow.getContents().get(i);
275                        Boolean symbolSecondRow = secondRow.getContents().get(i);
276                        if (!symbolFirstRow.equals(symbolSecondRow)) {
277                                return suffixes.get(i);
278                        }
279                }
280
281                throw new IllegalStateException("Both rows are identical, unable to determine a witness!");
282        }
283
284        @Nonnull
285        ObservationTableRow<I> getRowForPrefix(@Nonnull Word<I> state) {
286                for (ObservationTableRow<I> row : shortPrefixRows) {
287                        if (row.getLabel().equals(state)) {
288                                return row;
289                        }
290                }
291
292                for (ObservationTableRow<I> row : longPrefixRows) {
293                        if (row.getLabel().equals(state)) {
294                                return row;
295                        }
296                }
297
298                throw new IllegalArgumentException("Unable to find a row for '" + state + "'");
299        }
300
301        /**
302         * Moves a single row from long prefix rows to short prefix rows.
303         *
304         * @param longPrefix
305         *              A row which must be part of the long prefix rows and
306         *              should be moved to the short prefix rows.
307         */
308        void moveLongPrefixToShortPrefixes(@Nonnull Word<I> longPrefix) {
309                ObservationTableRow<I> rowToMove = null;
310
311                for (ObservationTableRow<I> row : longPrefixRows) {
312                        if (row.getLabel().equals(longPrefix)) {
313                                rowToMove = row;
314                                break;
315                        }
316                }
317
318                if (rowToMove == null) {
319                        throw new IllegalArgumentException("Word '" + longPrefix + "' not part of long prefixes");
320                }
321
322                longPrefixRows.remove(rowToMove);
323                rowToMove.setShortPrefixRow();
324                shortPrefixRows.add(rowToMove);
325        }
