079        private static class QueueElement<I,O> {
080                private Integer parentState;
081                private QueueElement<I,O> parentElement;
082                private I transIn;
083                private O transOut;
084                private int depth;
085
086                private QueueElement(Integer parentState, QueueElement<I,O> parentElement, I transIn, O transOut) {
087                        this.parentState = parentState;
088                        this.parentElement = parentElement;
089                        this.transIn = transIn;
090                        this.transOut = transOut;
091                        this.depth = (parentElement != null) ? parentElement.depth+1 : 0;
092                }

105        /**
106         * Constructor.
107         * @param alphabet the learning alphabet
108         * @param oracle the learning membership oracle
109         * @param suffixFinder the {@link GlobalSuffixFinder suffix finder} to use for analyzing counterexamples
110         * @param initialSuffixes the initial set of splitters, {@code null} or an empty collection will result
111         * in the set of splitters being initialized as the set of alphabet symbols (interpreted as {@link Word}s) 
112         */
113        public MealyDHC(Alphabet<I> alphabet,
114                        MembershipOracle<I, Word<O>> oracle,
115                        GlobalSuffixFinder<? super I, ? super Word<O>> suffixFinder,
116                        Collection<? extends Word<I>> initialSuffixes) {
117                this.alphabet = alphabet;
118                this.oracle = oracle;
119                this.suffixFinder = suffixFinder;
120                if(initialSuffixes == null || initialSuffixes.isEmpty()) {
121                        for(I symbol : alphabet) {
122                                splitters.add(Word.fromLetter(symbol));
123                        }
124                }
125                else {
126                        splitters.addAll(initialSuffixes);
127                }
128        }

130        /*
131         * (non-Javadoc)
132         * @see de.learnlib.api.LearningAlgorithm#startLearning()
133         */
134        @Override
135        public void startLearning() {
136                // initialize structure to store state output signatures
137                Map<List<Word<O>>, Integer> signatures = new HashMap<>();
138
139                // set up new hypothesis machine
140                hypothesis = new CompactMealy<>(alphabet);
141
142                // initialize exploration queue
143                Queue<QueueElement<I,O>> queue = new ArrayDeque<>();
144                
145                // initialize storage for access sequences
146                accessSequences = hypothesis.createDynamicStateMapping();
147
148                // first element to be explored represents the initial state with no predecessor
149                queue.add(new QueueElement<I,O>(null, null, null, null));
150
151                Interner<Word<O>> deduplicator = Interners.newStrongInterner();
152                
153                while (!queue.isEmpty()) {
154                        // get element to be explored from queue
155                        QueueElement<I,O> elem = queue.poll();
156
157                        // determine access sequence for state
158                        Word<I> access = assembleAccessSequence(elem);
159
160                        // assemble queries
161                        ArrayList<DefaultQuery<I, Word<O>>> queries = new ArrayList<>(splitters.size());
162                        for (Word<I> suffix : splitters) {
163                                queries.add(new DefaultQuery<I, Word<O>>(access, suffix));
164                        }
165
166                        // retrieve answers
167                        oracle.processQueries(queries);
168
169                        // assemble output signature
170                        List<Word<O>> sig = new ArrayList<>(splitters.size());
171                        for (DefaultQuery<I, Word<O>> query : queries) {
172                                sig.add(deduplicator.intern(query.getOutput()));
173                        }
174
175                        Integer sibling = signatures.get(sig);
176
177                        if (sibling != null) {
178                                // this element does not possess a new output signature
179                                // create a transition from parent state to sibling
180                                hypothesis.addTransition(elem.parentState, elem.transIn, sibling, elem.transOut);
181                        } else {
182                                // this is actually an observably distinct state! Progress!
183                                // Create state and connect via transition to parent
184                                Integer state = elem.parentElement == null ? hypothesis.addInitialState() : hypothesis.addState();
185                                if (elem.parentElement != null) {
186                                        hypothesis.addTransition(elem.parentState, elem.transIn, state, elem.transOut);
187                                }
188                                signatures.put(sig, state);
189                                accessSequences.put(state, elem);
190
191                                scheduleSuccessors(elem, state, queue, sig);
192                        }
193                }
194        }
195
196        private Word<I> assembleAccessSequence(QueueElement<I,O> elem) {
197                List<I> word = new ArrayList<>(elem.depth);
198                
199                QueueElement<I,O> pre = elem.parentElement;
200                I sym = elem.transIn;
201                while(pre != null && sym != null) {
202                        word.add(sym);
203                        sym = pre.transIn;
204                        pre = pre.parentElement;
205                }
206                
207                Collections.reverse(word);
208                return Word.fromList(word);
209        }
210
211        private void scheduleSuccessors(QueueElement<I,O> elem, Integer state, Queue<QueueElement<I,O>> queue, List<Word<O>> sig) throws IllegalArgumentException {
212                for (int i = 0; i < alphabet.size(); ++i) {
213                        // retrieve I/O for transition
214                        I input = alphabet.getSymbol(i);
215                        O output = sig.get(i).getSymbol(0);
216
217                        // create successor element and schedule for exploration
218                        queue.add(new QueueElement<>(state, elem, input, output));
219                }
220        }
221        
222        private void checkInternalState() {
223                if (hypothesis == null) {
224                        throw new IllegalStateException("No hypothesis learned yet");
225                }
226        }

228        /*
229         * (non-Javadoc)
230         * @see de.learnlib.api.LearningAlgorithm#refineHypothesis(de.learnlib.oracles.DefaultQuery)
231         */
232        @Override
233        public boolean refineHypothesis(DefaultQuery<I, Word<O>> ceQuery) {
234                checkInternalState();
235
236                Collection<? extends Word<I>> ceSuffixes
237                                = suffixFinder.findSuffixes(ceQuery, this, hypothesis, oracle);
238                
239                return addSuffixesUnchecked(ceSuffixes);
240        }

256        @Override
257        public Word<I> transformAccessSequence(Word<I> word) {
258                checkInternalState();
259                Integer state = hypothesis.getSuccessor(hypothesis.getInitialState(), word);
260                return assembleAccessSequence(accessSequences.get(state));
261        }

267        @Override

269                checkInternalState();

279        @Override
280        public boolean addGlobalSuffixes(Collection<? extends Word<I>> newGlobalSuffixes) {
281                checkInternalState();
282                
283                return addSuffixesUnchecked(newGlobalSuffixes);

285        
286        protected boolean addSuffixesUnchecked(Collection<? extends Word<I>> newSuffixes) {
287                int oldSize = hypothesis.size();
288                
289                for(Word<I> suf : newSuffixes) {
290                        if(!splitters.contains(suf)) {
291                                splitters.add(suf);
292                                log.log(Level.FINE, "added suffix: {0}", suf);
293                        }
294                }
295                
296                startLearning();
297                
298                return (hypothesis.size() != oldSize);
299        }
