017package de.learnlib.algorithms.lstargeneric.closing;
018
019import java.util.ArrayList;
020import java.util.List;
021
022import net.automatalib.commons.util.comparison.CmpUtil;
023import net.automatalib.words.Alphabet;
024import de.learnlib.algorithms.lstargeneric.table.ObservationTable;
025import de.learnlib.algorithms.lstargeneric.table.Row;
026import de.learnlib.api.MembershipOracle;
027
028
029/**
030 * Collection of predefined observation table closing strategies.
031 * 
032 * @see ClosingStrategy
033 * 
034 * @author Malte Isberner <malte.isberner@gmail.com>
035 *
036 */
037public class ClosingStrategies {
038        
039        /**
040         * Closing strategy that randomly selects one representative row to close from each equivalence
041         * class.
042         */
043        public static final ClosingStrategy<Object,Object> CLOSE_RANDOM
044                = new CloseRandomStrategy();
045        
046        /**
047         * Closing strategy that selects the first row from each equivalence class as representative.
048         */
049        public static final ClosingStrategy<Object,Object> CLOSE_FIRST
050                = new ClosingStrategy<Object,Object>() {
051                        @Override
052                        public <RI, RO> List<Row<RI>> selectClosingRows(
053                                        List<List<Row<RI>>> unclosedClasses,
054                                        ObservationTable<RI, RO> table,
055                                        MembershipOracle<RI, RO> oracle) {
056                                List<Row<RI>> result = new ArrayList<Row<RI>>(unclosedClasses.size());
057                                for(List<Row<RI>> clazz : unclosedClasses)
058                                        result.add(clazz.get(0));
059                                return result;
060                        }
061                        @Override
062                        public String toString() {
063                                return "CloseFirst";
064                        }
065        };
066        
067        /**
068         * Closing strategy that selects the shortest row of each equivalence class (more precisely:
069         * a row which's prefix has minimal length in the respective class) as representative. 
070         */
071        public static final ClosingStrategy<Object,Object> CLOSE_SHORTEST
072                = new ClosingStrategy<Object,Object>() {
073                        @Override
074                        public <RI, RO> List<Row<RI>> selectClosingRows(
075                                        List<List<Row<RI>>> unclosedClasses,
076                                        ObservationTable<RI, RO> table,
077                                        MembershipOracle<RI, RO> oracle) {
078                                
079                                List<Row<RI>> result = new ArrayList<Row<RI>>();
080                                for(List<Row<RI>> clazz : unclosedClasses) {
081                                        Row<RI> shortest = null;
082                                        int shortestLen = Integer.MAX_VALUE;
083                                        for(Row<RI> row : clazz) {
084                                                int prefixLen = row.getPrefix().length();
085                                                if(shortest == null || prefixLen < shortestLen) {
086                                                        shortest = row;
087                                                        shortestLen = prefixLen;
088                                                }
089                                        }
090                                        result.add(shortest);
091                                }
092                                return result;
093                        }
094                        @Override
095                        public String toString() {
096                                return "CloseShortest";
097                        }
098        };
099        
100        /**
101         * Closing strategy that selects the lexicographically minimal row (wrt. its prefix)
102         * of each equivalence class as representative.
103         */
104        public static final ClosingStrategy<Object,Object> CLOSE_LEX_MIN
105                = new ClosingStrategy<Object,Object>() {
106                        @Override
107                        public <RI, RO> List<Row<RI>> selectClosingRows(
108                                        List<List<Row<RI>>> unclosedClasses,
109                                        ObservationTable<RI, RO> table,
110                                        MembershipOracle<RI, RO> oracle) {
111                                List<Row<RI>> result = new ArrayList<Row<RI>>(unclosedClasses.size());
112                                Alphabet<RI> alphabet = table.getInputAlphabet();
113                                for(List<Row<RI>> clazz : unclosedClasses) {
114                                        Row<RI> lexMin = null;
115                                        for(Row<RI> row : clazz) {
116                                                if(lexMin == null)
117                                                        lexMin = row;
118                                                else if(CmpUtil.lexCompare(row.getPrefix(), lexMin.getPrefix(), alphabet) < 0)
119                                                        lexMin = row;
120                                        }
121                                        result.add(lexMin);
122                                }
123                                return result;
124                        }
125                        @Override
126                        public String toString() {
127                                return "CloseLexMin";
128                        }
129        };
130
131        @SuppressWarnings("unchecked")
132        public static ClosingStrategy<Object,Object>[] values() {
133                return new ClosingStrategy[]{
134                                CLOSE_RANDOM,
135                                CLOSE_FIRST,
136                                CLOSE_SHORTEST,
137                                CLOSE_LEX_MIN
138                        };
139        }
140
141}
