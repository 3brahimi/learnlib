129        /**
130         * Transforms a {@link LocalSuffixFinder} into a global one. Since local suffix finders
131         * only return a single suffix, suffix-closedness of the set of distinguishing suffixes
132         * might not be preserved. Note that for correctly implemented local suffix finders, this
133         * does not impair correctness of the learning algorithm. However, without suffix closedness,
134         * intermediate hypothesis models might be non-canonical, if no additional precautions
135         * are taken. For that reasons, the <tt>allSuffixes</tt> parameter can be specified to control
136         * whether or not the list returned by
137         * {@link GlobalSuffixFinder#findSuffixes(Query, AccessSequenceTransformer, SuffixOutput, MembershipOracle)}
138         * of the returned global suffix finder should not only contain the single suffix, but also
139         * all of its suffixes, ensuring suffix-closedness.
140         *  
141         * @param localFinder the local suffix finder
142         * @param allSuffixes whether or not all suffixes of the found local suffix should be added
143         * @return a global suffix finder using the analysis method from the specified local suffix finder
144         */
145        public static <I,O> GlobalSuffixFinder<I,O> fromLocalFinder(
146                        final LocalSuffixFinder<I,O> localFinder,
147                        final boolean allSuffixes) {
148                
149                return new GlobalSuffixFinder<I,O>() {
150                        @Override
151                        public <RI extends I,RO extends O>
152                        List<? extends Word<RI>> findSuffixes(Query<RI, RO> ceQuery,
153                                        AccessSequenceTransformer<RI> asTransformer,
154                                        SuffixOutput<RI, RO> hypOutput, MembershipOracle<RI, RO> oracle) {
155                                int idx = localFinder.findSuffixIndex(ceQuery, asTransformer, hypOutput, oracle);
156                                return suffixesForLocalOutput(ceQuery, idx, allSuffixes);
157                        }
158                        @Override
159                        public String toString() {
160                                return localFinder.toString() + (allSuffixes ? "-AllSuffixes" : "");
161                        }
162                };
163        }
164        
165        /**
166         * Transforms a {@link LocalSuffixFinder} into a global one. This is a convenience method,
167         * behaving like <tt>fromLocalFinder(localFinder, false)</tt>.
168         * @see #fromLocalFinder(LocalSuffixFinder, boolean)
169         */
170        public static <I,O> GlobalSuffixFinder<I,O> fromLocalFinder(LocalSuffixFinder<I,O> localFinder) {
171                return fromLocalFinder(localFinder, false);
172        }

175        /**
176         * Returns all suffixes of the counterexample word as distinguishing suffixes, as suggested
177         * by Mahler & Pnueli.
178         * @param ceQuery the counterexample query
179         * @return all suffixes of the counterexample input
180         */
181        public static <I,O> List<? extends Word<I>> findMahlerPnueli(Query<I,O> ceQuery) {
182                return ceQuery.getInput().suffixes(false);
183        }
184        
185        /**
186         * Returns all suffixes of the counterexample word as distinguishing suffixes, after
187         * stripping a maximal one-letter extension of an access sequence, as suggested by
188         * Shahbaz. 
189         * @param ceQuery the counterexample query
190         * @param asTransformer the access sequence transformer
191         * @return all suffixes from the counterexample after stripping a maximal one-letter
192         * extension of an access sequence. 
193         */
194        public static <I,O> List<? extends Word<I>> findShahbaz(Query<I,O> ceQuery,
195                        AccessSequenceTransformer<I> asTransformer) {
196                Word<I> queryWord = ceQuery.getInput();
197                int queryLen = queryWord.length();
198                
199                Word<I> prefix = ceQuery.getPrefix();
200                int i = prefix.length();
201                
202                while(i <= queryLen) {
203                        Word<I> nextPrefix = queryWord.prefix(i);
204                        
205                        if(!asTransformer.isAccessSequence(nextPrefix))
206                                break;
207                        i++;
208                }
209                
210                return queryWord.subWord(i).suffixes(false);
211        }

274        /**
275         * Transforms a suffix index returned by a {@link LocalSuffixFinder} into a list containing
276         * the single distinguishing suffix.
277         */
278        public static <I,O> List<? extends Word<I>> suffixesForLocalOutput(Query<I,O> ceQuery,
279                        int localSuffixIdx) {
280                return suffixesForLocalOutput(ceQuery, localSuffixIdx, false);
281        }
282        
283        /**
284         * Transforms a suffix index returned by a {@link LocalSuffixFinder} into a list of distinguishing
285         * suffixes. This list always contains the corresponding local suffix. Since local suffix finders
286         * only return a single suffix, suffix-closedness of the set of distinguishing suffixes
287         * might not be preserved. Note that for correctly implemented local suffix finders, this
288         * does not impair correctness of the learning algorithm. However, without suffix closedness,
289         * intermediate hypothesis models might be non-canonical, if no additional precautions
290         * are taken. For that reasons, the <tt>allSuffixes</tt> parameter can be specified to control
291         * whether or not the list returned by
292         * {@link GlobalSuffixFinder#findSuffixes(Query, AccessSequenceTransformer, SuffixOutput, MembershipOracle)}
293         * of the returned global suffix finder should not only contain the single suffix, but also
294         * all of its suffixes, ensuring suffix-closedness.
295         */
296        public static <I,O> List<? extends Word<I>> suffixesForLocalOutput(Query<I,O> ceQuery,
297                        int localSuffixIdx, boolean allSuffixes) {
298                
299                if(localSuffixIdx == -1)
300                        return Collections.emptyList();
301                
302                Word<I> suffix = ceQuery.getInput().subWord(localSuffixIdx);
303                
304                if(!allSuffixes)
305                        return Collections.singletonList(suffix);
306                
307                return suffix.suffixes(false);
308        }
