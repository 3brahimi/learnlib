017package de.learnlib.oracles;
018
019import javax.annotation.Nonnull;
020import javax.annotation.Nullable;
021import javax.annotation.ParametersAreNonnullByDefault;
022
023import de.learnlib.api.SUL;
024import de.learnlib.statistics.Counter;
025import de.learnlib.statistics.StatisticSUL;
026
027@ParametersAreNonnullByDefault
028public class SymbolCounterSUL<I, O> implements StatisticSUL<I,O> {
029        
030        private final SUL<I,O> sul;
031        private final Counter counter;
032
033        public SymbolCounterSUL(String name, SUL<I,O> sul) {
034                this.sul = sul;
035                this.counter = new Counter(name, "symbols");
036        }
037
038        /* (non-Javadoc)
039         * @see de.learnlib.api.SUL#pre()
040         */
041        @Override
042        public void pre() {
043                sul.pre();
044        }
045
046
047        /* (non-Javadoc)
048         * @see de.learnlib.api.SUL#post()
049         */
050        @Override
051        public void post() {
052                sul.post();
053        }
054
055        /* (non-Javadoc)
056         * @see de.learnlib.api.SUL#step(java.lang.Object)
057         */
058        @Override
059        @Nullable
060        public O step(@Nullable I in) {
061                counter.increment();
062                return sul.step(in);
063        }
064
065        /* (non-Javadoc)
066         * @see de.learnlib.statistics.StatisticSUL#getStatisticalData()
067         */
068        @Override
069        @Nonnull
070        public Counter getStatisticalData() {
071                return counter;
072        }
073        
074        
075
076}
