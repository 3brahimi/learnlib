017package de.learnlib.api;
018
019import javax.annotation.Nonnull;
020
021import net.automatalib.automata.fsa.DFA;
022import net.automatalib.automata.transout.MealyMachine;
023import net.automatalib.words.Word;
024import de.learnlib.oracles.DefaultQuery;
025
026
027
028/**
029 * Basic interface for a model inference algorithm.
030 * <p>
031 * Actively inferring models (such as DFAs or Mealy machines) consists of the construction
032 * of an initial hypothesis, which is subsequently refined using counterexamples
033 * (see {@link EquivalenceOracle}). 
034 * 
035 * @author Maik Merten <maikmerten@googlemail.com>
036 * @author Malte Isberner <malte.isberner@gmail.com>
037 *
038 * @param <M> model class.
039 * @param <I> input symbol class.
040 * @param <O> output class.
041 */
042public interface LearningAlgorithm<M, I, O> {
043        
044        public static interface DFALearner<I> extends LearningAlgorithm<DFA<?,I>,I,Boolean> {}
045        public static interface MealyLearner<I,O> extends LearningAlgorithm<MealyMachine<?,I,?,O>,I,Word<O>> {}
046        
047        /**
048         * Starts the model inference process, creating an initial hypothesis in the provided
049         * model object. Please note that it should be illegal to invoke this method twice.
050         */
051        public void startLearning();
052        
053        /**
054         * Triggers a refinement of the model by providing a counterexample.
055         * A counterexample is a query which exposes different behavior of the real SUL compared
056         * to the hypothesis. Please note that invoking this method before an initial
057         * invocation of {@link #startLearning()} should be illegal.
058         * 
059         * @param ceQuery the query which exposes diverging behavior, as posed to the real SUL
060         * (i.e. with the SULs output).
061         * @return <tt>true</tt> if the counterexample triggered a refinement of the hypothesis,
062         * <tt>false</tt> otherwise (i.e., it was no counterexample).
063         */
064        public boolean refineHypothesis(@Nonnull DefaultQuery<I, O> ceQuery);
065        
066        /**
067         * Returns the current hypothesis model.
068         * <p>
069         * N.B.: By the contract of this interface, the model returned by this method may not be
070         * modified (i.e., M generally should refer to an immutable interface), and its validity
071         * is retained only until the next invocation of {@link #refineHypothesis(DefaultQuery)}. If
072         * older hypotheses have to be maintained, a copy of the returned model must be made
073         * Please note that it should be illegal to invoke this method before an initial invocation
074         * of {@link #startLearning()}.
075         * @return the current hypothesis model.
076         */
077        @Nonnull
078        public M getHypothesisModel();
079}
