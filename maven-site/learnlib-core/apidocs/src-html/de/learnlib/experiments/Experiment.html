017package de.learnlib.experiments;
018
019import net.automatalib.words.Alphabet;
020import de.learnlib.api.EquivalenceOracle;
021import de.learnlib.api.LearningAlgorithm;
022import de.learnlib.logging.LearnLogger;
023import de.learnlib.oracles.DefaultQuery;
024import de.learnlib.statistics.Counter;
025import de.learnlib.statistics.SimpleProfiler;
026
027/**
028 * runs a learning experiment. 
029 * 
030 * @author falkhowar
031 * @param <A>
032 * @param <I>
033 * @param <O> 
034 */
035public class Experiment<A> {
036        
037        private final class ExperimentImpl<I,O> {
038                private final LearningAlgorithm<? extends A, I, O> learningAlgorithm;
039            private final EquivalenceOracle<? super A, I, O> equivalenceAlgorithm;
040            private final Alphabet<I> inputs;
041            
042            public ExperimentImpl(LearningAlgorithm<? extends A, I, O> learningAlgorithm, EquivalenceOracle<? super A, I, O> equivalenceAlgorithm, Alphabet<I> inputs) {
043                this.learningAlgorithm = learningAlgorithm;
044                this.equivalenceAlgorithm = equivalenceAlgorithm;
045                this.inputs = inputs;
046            }
047            
048            public A run() {
049                rounds.increment();
050                logger.logPhase("Starting round " + rounds.getCount());
051                logger.logPhase("Learning");
052                profileStart("Learning");
053                learningAlgorithm.startLearning();
054                profileStop("Learning");
055
056                boolean done = false;
057                A hyp = null;
058                while (!done) {
059                        hyp = learningAlgorithm.getHypothesisModel();
060                    if (logModels) {
061                        logger.logModel(hyp);
062                    }
063
064                    logger.logPhase("Searching for counterexample");
065                    profileStart("Searching for counterexample");
066                    DefaultQuery<I, O> ce = equivalenceAlgorithm.findCounterExample(hyp, inputs);
067                    if (ce == null) {
068                        done = true;
069                        continue;
070                    }
071                    profileStop("Searching for counterexample");
072                    
073                    logger.logCounterexample(ce.getInput().toString());
074
075                    // next round ...
076                    rounds.increment();
077                    logger.logPhase("Starting round " + rounds.getCount());
078                    logger.logPhase("Learning");
079                    profileStart("Learning");
080                    learningAlgorithm.refineHypothesis(ce);
081                    profileStop("Learning");
082                }
083
084                return hyp;
085            }
086        }
087
088    private static LearnLogger logger = LearnLogger.getLogger(Experiment.class);
089    
090    private boolean logModels = false;
091    private boolean profile = false;
092    private Counter rounds = new Counter("rounds", "#");
093    private A finalHypothesis = null;
094    private final ExperimentImpl<?,?> impl;
095
096    public <I,O> Experiment(LearningAlgorithm<? extends A, I, O> learningAlgorithm, EquivalenceOracle<? super A, I, O> equivalenceAlgorithm, Alphabet<I> inputs) {
097        this.impl = new ExperimentImpl<>(learningAlgorithm, equivalenceAlgorithm, inputs);
098    }
099    
100
101    
102    /**
103     * 
104     */
105    public A run() {
106        finalHypothesis = impl.run();
107        return finalHypothesis;
108    }
109    
110    public A getFinalHypothesis() {
111        return finalHypothesis;
112    }
113
114    
115    
116    private void profileStart(String taskname) {
117        if (profile) {
118            SimpleProfiler.start(taskname);
119        }
120    }
121
122    private void profileStop(String taskname) {
123        if (profile) {
124            SimpleProfiler.stop(taskname);
125        }
126    }
127
128    /**
129     * @param logModels the logModels to set
130     */
131    public void setLogModels(boolean logModels) {
132        this.logModels = logModels;
133    }
134
135    /**
136     * @param profile the profile to set
137     */
138    public void setProfile(boolean profile) {
139        this.profile = profile;
140    }
141
142    /**
143     * @return the rounds
144     */
145    public Counter getRounds() {
146        return rounds;
147    }
148}
