049/**
050 * This example shows how a model of a Java class can be learned using the SUL
051 * (system under learning) interfaces and the random walks equivalence test.
052 *
053 * @author falkhowar
054 */
055public class Example {
056
057    /*
058     * The BoundedStringQueue is the class of which we are going to 
059     * infer a model. It wraps an ordinary queue of Strings, limiting
060     * its size to MAX_SIZE (3). Once the queue is full, additional 
061     * offers will be ignored.
062     * <p>
063     * However, the implementation uses the underlying queue in a strange
064     * way as the model will reveal.
065     */
066    public static class BoundedStringQueue {
067
068        // capacity
069        public static final int MAX_SIZE = 3;
070        // storage
071        private Deque<String> data = new ArrayDeque<>(3);
072
073        // add a String to the queue if capacity allows
074        public void offer(String s) {
075            if (data.size() < MAX_SIZE) {
076                data.offerFirst(s);
077            }
078        }
079
080        // get next element from queue (null for empty queue)
081        public String poll() {
082            return data.poll();
083        }
084    }
085
086    /*
087     * We use the BSQInput class to wrap concrete method invocations
088     * and use these as alphabet symbols for the learning algorithm
089     */
090    public static class BSQInput {
091
092        // method to invoke
093        public final Method action;
094        
095        // method parameter values
096        public final Object[] data;
097
098        public BSQInput(Method action, Object[] data) {
099            this.action = action;
100            this.data = data;
101        }
102
103        // this will be used for printing when 
104        // logging or exporting automata
105        @Override
106        public String toString() {
107            return action.getName() + Arrays.toString(data);
108        }
109    }
110
111    /*
112     * The BSQAdapter 
113     * 
114     */
115    public static class BSQAdapter implements SUL<BSQInput, String> {
116
117        // system under learning
118        private BoundedStringQueue sul;
119
120        // reset the SUL
121        @Override
122        public void reset() {
123            // we just create a new instance
124            sul = new BoundedStringQueue();
125        }
126
127        // execute one input on the SUL
128        @Override
129        public String step(BSQInput in) {
130            try {
131                // invoke the method wrapped by in
132                Object ret = in.action.invoke(sul, in.data);
133                // make sure that we return a string
134                return ret == null ? "" : (String) ret;
135            } 
136            catch (IllegalAccessException | 
137                    IllegalArgumentException | 
138                    InvocationTargetException e) {
139                // This should never happen. In a real experiment
140                // this would be the point when we want to issue
141                // a warning or stop the learning.
142                return "err";
143            }
144        }
145    }
146
147    public static void main(String[] args) throws NoSuchMethodException, IOException {
148
149        // create learning alphabet
150        
151        // offer(a)
152        BSQInput offer_a = new BSQInput(BoundedStringQueue.class.getMethod(
153                "offer", new Class<?>[]{String.class}), new Object[]{"a"});
154
155        // offer(b)
156        BSQInput offer_b = new BSQInput(BoundedStringQueue.class.getMethod(
157                "offer", new Class<?>[]{String.class}), new Object[]{"b"});
158
159        // poll()
160        BSQInput poll = new BSQInput(BoundedStringQueue.class.getMethod(
161                "poll", new Class<?>[]{}), new Object[]{});
162
163        Alphabet<BSQInput> inputs = new SimpleAlphabet<>();
164        inputs.add(offer_a);
165        inputs.add(offer_b);
166        inputs.add(poll);
167
168        // create an oracle that can answer membership queries
169        // using the BSQAdapter
170        SUL<BSQInput, String> sul = new BSQAdapter();
171        SULOracle<BSQInput, String> backOracle = new SULOracle<>(sul);
172
173        // oracle for counting queries wraps sul
174        CounterOracle<BSQInput, Word<String>> mqOracle =
175                new CounterOracle<>(backOracle, "membership queries");
176
177        // create initial set of suffixes
178        List<Word<BSQInput>> suffixes = new ArrayList<>();
179        suffixes.add(Word.fromSymbols(offer_a));
180        suffixes.add(Word.fromSymbols(offer_b));
181        suffixes.add(Word.fromSymbols(poll));
182
183        // construct L* instance (almost classic Mealy version)
184        // almost: we use words (Word<String>) in cells of the table 
185        // instead of single outputs.
186        LearningAlgorithm<? extends MealyMachine<?, BSQInput, ?, String>, BSQInput, Word<String>> lstar =
187                new ExtensibleLStarMealy<>(
188                inputs, // input alphabet
189                mqOracle, // mq oracle
190                suffixes, // initial suffixes
191                ObservationTableCEXHandlers.CLASSIC_LSTAR, // handling of counterexamples
192                ClosingStrategies.CLOSE_FIRST // always choose first unclosedness found 
193                );
194
195        // create random walks equivalence test
196        EquivalenceOracle<MealyMachine<?, BSQInput, ?, String>, BSQInput, Word<String>> randomWalks =
197                new RandomWalkEQOracle<>(
198                0.05, // reset SUL w/ this probability before a step 
199                10000, // max steps (overall)
200                false, // reset step count after counterexample 
201                new Random(46346293), // make results reproducible 
202                sul // system under learning
203                );
204
205        // construct a learning experiment from
206        // the learning algorithm and the random walks test.
207        // The experiment will execute the main loop of
208        // active learning
209        Experiment<MealyMachine<?,BSQInput,?,String>> experiment =
210                new Experiment<>(lstar, randomWalks, inputs);
211
212        // turn on time profiling
213        experiment.setProfile(true);
214
215        // enable logging of models
216        experiment.setLogModels(true);
217
218        // run experiment
219        experiment.run();
220
221        // get learned model
222        MealyMachine<?, BSQInput, ?, String> result = experiment.getFinalHypothesis();
223
224        // report results
225        System.out.println("-------------------------------------------------------");
226
227        // profiling
228        System.out.println(SimpleProfiler.getResults());
229
230        // learning statistics
231        System.out.println(experiment.getRounds().getSummary());
232        System.out.println(mqOracle.getCounter().getSummary());
233
234        // model statistics
235        System.out.println("States: " + result.size());
236        System.out.println("Sigma: " + inputs.size());
237
238        // show model
239        System.out.println();
240        System.out.println("Model: ");
241        
242        GraphDOT.write(result, inputs, System.out); // may throw IOException!
243        Writer w = DOT.createDotWriter(true);
244        GraphDOT.write(result, inputs, w);
245        w.close();
246
247        System.out.println("-------------------------------------------------------");
248
249    }
