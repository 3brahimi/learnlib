017package de.learnlib.parallelism;
018
019import java.util.Arrays;
020import java.util.Collection;
021import java.util.Iterator;
022import java.util.List;
023import java.util.concurrent.CountDownLatch;
024
025import de.learnlib.api.MembershipOracle;
026import de.learnlib.api.Query;
027
028
029/**
030 * A membership oracle that distributes a set of queries among several threads.
031 * 
032 * @author Malte Isberner <malte.isberner@gmail.com>
033 *
034 * @param <I> input symbol class
035 * @param <O> output class
036 */
037public class ParallelOracle<I, O> implements MembershipOracle<I, O> {
038        
039        // TODO Does this number make sense?
040        public static int DEFAULT_MIN_BATCH_SIZE = 10;
041        
042        private final int minBatchSize;
043        private final OracleWorker<I, O>[] workers;
044        private final Thread[] workerThreads;
045        private final MembershipOracle<I,O> thisThreadOracle;
046
047        /**
048         * Constructor for using (potentially) <i>separate</i> oracles for each worker thread,
049         * with a default minimum batch size.
050         * @param oracles the oracles to use for answering the queries. The cardinality of this
051         * list coincides with the number of threads created.
052         */
053        public ParallelOracle(List<? extends MembershipOracle<I,O>> oracles) {
054                this(oracles, DEFAULT_MIN_BATCH_SIZE);
055        }
056        
057        /**
058         * Constructor for using (potentially) <i>separate</i> oracles for each worker thread.
059         * @param oracles the oracles to use for answering the queries. The cardinality of this
060         * list coincides with the number of threads created.
061         * @param minBatchSize the minimum batch size
062         */
063        @SuppressWarnings("unchecked")
064        public ParallelOracle(List<? extends MembershipOracle<I,O>> oracles, int minBatchSize) {
065                int numOracles = oracles.size();
066                if(numOracles <= 0)
067                        throw new IllegalArgumentException("Must provide at least one oracle");
068                workers = new OracleWorker[numOracles - 1];
069                workerThreads = new Thread[numOracles - 1];
070                Iterator<? extends MembershipOracle<I,O>> it = oracles.iterator();
071                thisThreadOracle = it.next();
072                int i = 0;
073                while(it.hasNext()) {
074                        MembershipOracle<I,O> oracle = it.next();
075                        OracleWorker<I,O> worker = new OracleWorker<>(oracle);
076                        workers[i++] = worker;
077                }
078                this.minBatchSize = minBatchSize;
079        }
080        
081        /**
082         * Constructor for using one <i>shared</i> oracle for all worker threads,
083         * with the default minimum batch size.
084         * @param sharedOracle the shared oracle
085         * @param numInstances the number of threads to create
086         */
087        public ParallelOracle(MembershipOracle<I,O> sharedOracle, int numInstances) {
088                this(sharedOracle, numInstances, DEFAULT_MIN_BATCH_SIZE);
089        }
090        
091        /**
092         * Constructor for using one <i>shared</i> oracle for all worker threads,
093         * with a custom minimum batch size.
094         * @param sharedOracle the shared oracle
095         * @param numInstances the number of threads to create
096         * @param minBatchSize the minimum batch size
097         */
098        @SuppressWarnings("unchecked")
099        public ParallelOracle(MembershipOracle<I,O> sharedOracle, int numInstances, int minBatchSize) {
100                if(numInstances <= 0)
101                        throw new IllegalArgumentException("Must have at least one oracle instance");
102                numInstances--;
103                workers = new OracleWorker[numInstances];
104                workerThreads = new Thread[numInstances];
105                thisThreadOracle = sharedOracle;
106                for(int i = 0; i < numInstances; i++) {
107                        OracleWorker<I,O> worker = new OracleWorker<>(sharedOracle);
108                        workers[i] = worker;
109                }
110                this.minBatchSize = minBatchSize;
111        }
112        
113        /**
114         * Starts all worker threads.
115         */
116        public void start() {
117                if(workerThreads.length > 0 && workerThreads[0] != null)
118                        throw new IllegalStateException("ParallelOracle already started");
119                
120                for(int i = 0; i < workers.length; i++) {
121                        Thread t = new Thread(workers[i]);
122                        workerThreads[i] = t;
123                        t.start();
124                }
125        }
126        
127        /*
128         * (non-Javadoc)
129         * @see de.learnlib.api.MembershipOracle#processQueries(java.util.Collection)
130         */
131        @Override
132        @SuppressWarnings("unchecked")
133        public void processQueries(Collection<? extends Query<I, O>> queries) {
134                if(workerThreads.length > 0 && workerThreads[0] == null)
135                        throw new IllegalStateException("ParallelOracle was not started");
136                
137                int num = queries.size();
138                if(num <= 0)
139                        return;
140                
141                int numBatches = (num - minBatchSize)/minBatchSize + 1;
142                if(numBatches > workers.length + 1)
143                        numBatches = workers.length + 1;
144                
145                // Calculate the number of full and non-full batches. The difference in size
146                // will never exceed one (cf. pidgeonhole principle)
147                int fullBatchSize = (num - 1)/numBatches + 1;
148                int nonFullBatches = fullBatchSize*numBatches - num;
149                
150                Iterator<? extends Query<I,O>> queryIt = queries.iterator();
151                
152                // One batch is always executed in the local thread. This saves the thread creation
153                // overhead for the common case where the batch size is quite small.
154                int externalBatches = numBatches - 1;
155                
156                // If we decide not to need any external threads, we can save initializing synchronization
157                // measures.
158                CountDownLatch finishSignal = (externalBatches > 0) ? new CountDownLatch(externalBatches) : null;
159                
160                // Start the threads for the external batches
161                for(int i = 0; i < externalBatches; i++) {
162                        int bs = fullBatchSize;
163                        if(i < nonFullBatches)
164                                bs--;
165                        Query<I,O>[] batch = new Query[bs];
166                        for(int j = 0; j < bs; j++)
167                                batch[j] = queryIt.next();
168                        
169                        workers[i].offerBatch(batch, finishSignal);
170                }
171                
172                // Finally, prepare and process the batch for the oracle executed in this thread.
173                Query<I,O>[] batch = new Query[fullBatchSize];
174                for(int j = 0; j < fullBatchSize; j++)
175                        batch[j] = queryIt.next();
176                
177                thisThreadOracle.processQueries(Arrays.asList(batch));
178                
179                // FIXME: Needs deadlock prevention
180                if(finishSignal != null) {
181                        try {
182                                finishSignal.await();
183                        } catch (InterruptedException e) {
184                                throw new IllegalStateException(e);
185                        }
186                }
187        }
188        
189        /**
190         * Stop all worker threads. After this method has been called, invoking {@link #processQueries(Collection)}
191         * will result in an {@link IllegalStateException} until the next call to {@link #start()}
192         * is made. 
193         */
194        public void stop() {
195                if(workerThreads.length > 0 && workerThreads[0] == null)
196                        throw new IllegalStateException("Parallel oracle was not started");
197                
198                for(int i = 0; i < workers.length; i++) {
199                        workers[i].stop();
200                }
201                for(int i = 0; i < workerThreads.length; i++) {
202                        try {
203                                workerThreads[i].join();
204                        } catch (InterruptedException e) {
205                                // TODO Auto-generated catch block
206                                e.printStackTrace();
207                        }
208                        workerThreads[i] = null;
209                }
210        }
