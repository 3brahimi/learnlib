018package de.learnlib.examples.mealy;
019
020import net.automatalib.automata.transout.impl.FastMealy;
021import net.automatalib.automata.transout.impl.FastMealyState;
022import net.automatalib.words.Alphabet;
023import net.automatalib.words.impl.FastAlphabet;
024import net.automatalib.words.impl.Symbol;
025
026/**
027 * This class provides the example used in the paper ''Inferring Mealy Machines'' 
028 * by Muzammil Shahbaz and Roland Groz (see Figure 1).
029 * 
030 * @author Oliver Bauer <oliver.bauer@tu-dortmund.de>
031 */
032public class ExampleShahbazGroz {
033        private final static Symbol in_a = new Symbol("a");
034    private final static Symbol in_b = new Symbol("b");
035    private final static Alphabet<Symbol> alphabet = new FastAlphabet<>(in_a, in_b);
036    
037    private final static String out_x = "x";
038    private final static String out_y = "y";
039    
040    /**
041     * Construct and return a machine representation of this example
042     * 
043     * @return machine instance of the example
044     */
045    public static FastMealy<Symbol, String> constructMachine() {
046        
047        FastMealy<Symbol, String> fm = new FastMealy<>(alphabet);
048        
049        FastMealyState<String> q0 = fm.addInitialState();
050        FastMealyState<String> q1 = fm.addState();
051        FastMealyState<String> q2 = fm.addState();
052        FastMealyState<String> q3 = fm.addState();
053        
054        fm.addTransition(q0, in_a, q1, out_x);
055        fm.addTransition(q0, in_b, q3, out_x);
056        
057        fm.addTransition(q1, in_a, q1, out_y);
058        fm.addTransition(q1, in_b, q2, out_x);
059        
060        fm.addTransition(q2, in_a, q3, out_x);
061        fm.addTransition(q2, in_b, q3, out_x);
062        
063        fm.addTransition(q3, in_a, q0, out_x);
064        fm.addTransition(q3, in_b, q0, out_x);
065        
066        /*
067         * In the paper the authors use the following counterexample
068         * to refine the first conjecture from an angluin for mealy machines:
069         * in_a in_b in_a in_b in_b in_a in_a
070         */
071        
072        return fm;
073    }
074}
