017package de.learnlib.examples.example1;
018
019import java.io.IOException;
020import java.io.Writer;
021import java.util.Collections;
022
023import net.automatalib.automata.fsa.DFA;
024import net.automatalib.commons.dotutil.DOT;
025import net.automatalib.util.graphs.dot.GraphDOT;
026import net.automatalib.words.Alphabet;
027import net.automatalib.words.Word;
028import net.automatalib.words.impl.Symbol;
029import de.learnlib.algorithms.lstargeneric.ce.ObservationTableCEXHandlers;
030import de.learnlib.algorithms.lstargeneric.closing.ClosingStrategies;
031import de.learnlib.algorithms.lstargeneric.dfa.ExtensibleLStarDFA;
032import de.learnlib.eqtests.basic.WMethodEQOracle;
033import de.learnlib.experiments.Experiment;
034import de.learnlib.oracles.CounterOracle;
035import de.learnlib.oracles.SimulatorOracle;
036import de.learnlib.statistics.SimpleProfiler;
037import net.automatalib.automata.fsa.impl.FastDFA;
038import net.automatalib.automata.fsa.impl.FastDFAState;
039import net.automatalib.words.impl.FastAlphabet;
040
041/**
042 * This example shows the usage of a learning algorithm and an equivalence test
043 * as part of an experiment in order to learn a simulated SUL (system under
044 * learning).
045 *
046 * @author falkhowar
047 */
048public class Example {
049
050    /**
051     * creates example from Angluin's seminal paper.
052     * 
053     * @return example dfa
054     */
055    private static FastDFA<Symbol> constructSUL() {
056        // create inputs
057        Symbol a = new Symbol("a");
058        Symbol b = new Symbol("b");
059        Alphabet<Symbol> sigma = new FastAlphabet<>(a, b);
060
061        // create states
062        FastDFA<Symbol> dfa = new FastDFA<>(sigma);
063        FastDFAState q0 = dfa.addInitialState(true);
064        FastDFAState q1 = dfa.addState(false);
065        FastDFAState q2 = dfa.addState(false);
066        FastDFAState q3 = dfa.addState(false);
067
068        // create transitions
069        dfa.addTransition(q0, a, q1);
070        dfa.addTransition(q0, b, q2);
071        dfa.addTransition(q1, a, q0);
072        dfa.addTransition(q1, b, q3);
073        dfa.addTransition(q2, a, q3);
074        dfa.addTransition(q2, b, q0);
075        dfa.addTransition(q3, a, q2);
076        dfa.addTransition(q3, b, q1);
077
078        return dfa;
079    }
080
081    public static void main(String[] args) throws IOException {
082
083        // load DFA and alphabet
084        // the ? leaves open the state implementation since we do not
085        // need to know it explicitly
086        FastDFA<Symbol> target = constructSUL();
087        Alphabet<Symbol> inputs = target.getInputAlphabet();
088
089        // typed empty word
090        Word<Symbol> epsilon = Word.epsilon();
091
092        // construct a simulator membership query oracle
093        // input  - Symbol  (determined by example)
094        // output - Boolean (determined by DFA)
095        SimulatorOracle<Symbol, Boolean> sul = new SimulatorOracle<>(target);
096
097        // oracle for counting queries wraps SUL
098        CounterOracle<Symbol, Boolean> mqOracle =
099                new CounterOracle<>(sul, "membership queries");
100
101        // construct L* instance
102        ExtensibleLStarDFA<Symbol> lstar = new ExtensibleLStarDFA<>(
103                inputs, // input alphabet
104                mqOracle, // mq oracle
105                Collections.singletonList(epsilon), // initial suffixes
106                ObservationTableCEXHandlers.CLASSIC_LSTAR, // handling of counterexamples
107                ClosingStrategies.CLOSE_FIRST // always choose first unclosedness found
108                );
109
110        // construct a W-method conformance test
111        // exploring the system up to depth 4 from
112        // every state of a hypothesis
113        WMethodEQOracle<DFA<?, Symbol>, Symbol, Boolean> wMethod =
114                new WMethodEQOracle<>(4, mqOracle);
115
116        // construct a learning experiment from
117        // the learning algorithm and the conformance test.
118        // The experiment will execute the main loop of
119        // active learning
120        Experiment<DFA<?, Symbol>> experiment =
121                new Experiment<>(lstar, wMethod, inputs);
122
123        // turn on time profiling
124        experiment.setProfile(true);
125
126        // enable logging of models
127        experiment.setLogModels(true);
128
129        // run experiment
130        experiment.run();
131
132        // get learned model
133        DFA<?, Symbol> result = experiment.getFinalHypothesis();
134
135        // report results
136        System.out.println("-------------------------------------------------------");
137
138        // profiling
139        System.out.println(SimpleProfiler.getResults());
140
141        // learning statistics
142        System.out.println(experiment.getRounds().getSummary());
143        System.out.println(mqOracle.getCounter().getSummary());
144
145        // model statistics
146        System.out.println("States: " + result.size());
147        System.out.println("Sigma: " + inputs.size());
148
149        // show model
150        System.out.println();
151        System.out.println("Model: ");
152        GraphDOT.write(result, inputs, System.out); // may throw IOException!
153
154        Writer w = DOT.createDotWriter(true);
155        GraphDOT.write(result, inputs, w);
156        w.close();
157
158        System.out.println("-------------------------------------------------------");
159    }
160}
