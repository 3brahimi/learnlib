001package de.learnlib.mealy;
002
003import java.util.Objects;
004
005import net.automatalib.automata.transout.MealyMachine;
006import net.automatalib.words.Word;
007import de.learnlib.api.LearningAlgorithm;
008import de.learnlib.api.MembershipOracle;
009import de.learnlib.oracles.DefaultQuery;
010
011
012/**
013 * Utility class helping to unify various approaches to actively learning
014 * Mealy machines.
015 *  
016 * @author Malte Isberner <malte.isberner@gmail.com>
017 *
018 */
019public abstract class MealyUtil {
020        
021        public static final int NO_MISMATCH = -1;
022        
023        public static <O> int findMismatch(Word<O> out1, Word<O> out2) {
024                int len = out1.length();
025                assert len == out2.length();
026                
027                for(int i = 0; i < len; i++) {
028                        O sym1 = out1.getSymbol(i);
029                        O sym2 = out2.getSymbol(i);
030                        
031                        if(!Objects.equals(sym1, sym2))
032                                return i;
033                }
034                
035                return NO_MISMATCH;
036        }
037        
038        
039        public static <I,O> DefaultQuery<I,Word<O>> shortenCounterExample(
040                        MealyMachine<?,I,?,O> hypothesis,
041                        DefaultQuery<I,Word<O>> ceQuery) {
042                Word<I> cePrefix = ceQuery.getPrefix(), ceSuffix = ceQuery.getSuffix();
043                Word<O> hypOut = hypothesis.computeSuffixOutput(cePrefix, ceSuffix);
044                Word<O> ceOut = ceQuery.getOutput();
045                assert ceOut.length() == hypOut.length();
046                
047                int mismatchIdx = findMismatch(hypOut, ceOut);
048                if(mismatchIdx == NO_MISMATCH)
049                        return null;
050                
051                return new DefaultQuery<>(cePrefix,
052                                ceSuffix.prefix(mismatchIdx + 1),
053                                ceOut.prefix(mismatchIdx + 1));
054        }
055        
056        public static <I,O> DefaultQuery<I,O> reduceCounterExample(
057                        MealyMachine<?,I,?,O> hypothesis,
058                        DefaultQuery<I,Word<O>> ceQuery) {
059                Word<I> cePrefix = ceQuery.getPrefix(), ceSuffix = ceQuery.getSuffix();
060                Word<O> hypOut = hypothesis.computeSuffixOutput(cePrefix, ceSuffix);
061                Word<O> ceOut = ceQuery.getOutput();
062                assert ceOut.length() == hypOut.length();
063                
064                int mismatchIdx = findMismatch(hypOut, ceOut);
065                if(mismatchIdx == NO_MISMATCH)
066                        return null;
067                
068                return new DefaultQuery<>(cePrefix,
069                                ceSuffix.prefix(mismatchIdx + 1),
070                                ceOut.getSymbol(mismatchIdx));
071        }
072        
073        
074        public static <M extends MealyMachine<?,I,?,O>,I,O>
075        LearningAlgorithm<M,I,Word<O>> wrapSymbolLearner(LearningAlgorithm<M,I,O> learner) {
076                return new MealyLearnerWrapper<>(learner);
077        }
078        
079        public static <I,O> MembershipOracle<I, O> wrapWordOracle(MembershipOracle<I,Word<O>> oracle) {
080                return new SymbolOracleWrapper<>(oracle);
081        }
082        
083        // Prevent inheritance
084        private MealyUtil() {}
085
086}
