020import java.util.Collections;

023import java.util.List;
024import java.util.Map;
025import java.util.Queue;
026import java.util.logging.Level;
027import java.util.logging.Logger;
028
029import net.automatalib.automata.transout.MealyMachine;
030import net.automatalib.automata.transout.impl.compact.CompactMealy;
031import net.automatalib.commons.util.mappings.MutableMapping;
032import net.automatalib.words.Alphabet;
033import net.automatalib.words.Word;
034import net.automatalib.words.impl.SimpleAlphabet;
035import de.learnlib.algorithms.dhc.Deduplicator;
036import de.learnlib.api.AccessSequenceTransformer;
037import de.learnlib.api.LearningAlgorithm;
038import de.learnlib.api.MembershipOracle;

043/**
044 *
045 * @author Maik Merten <maikmerten@googlemail.com>
046 */
047public class MealyDHC<I, O> implements LearningAlgorithm<MealyMachine<?, I, ?, O>, I, Word<O>>,
048                AccessSequenceTransformer<I> {
049
050        private static final Logger log = Logger.getLogger( MealyDHC.class.getName() );
051        
052        private Alphabet<I> alphabet;
053        private MembershipOracle<I, Word<O>> oracle;
054        private SimpleAlphabet<Word<I>> splitters = new SimpleAlphabet<>();
055        private CompactMealy<I, O> hypothesis;
056        private MutableMapping<Integer, QueueElement<I,O>> accessSequences;
057        private GlobalSuffixFinder<? super I,? super Word<O>> suffixFinder;
058
059
060        private static class QueueElement<I,O> {
061                private Integer parentState;
062                private QueueElement<I,O> parentElement;
063                private I transIn;
064                private O transOut;
065                private int depth;
066
067                private QueueElement(Integer parentState, QueueElement<I,O> parentElement, I transIn, O transOut) {
068                        this.parentState = parentState;
069                        this.parentElement = parentElement;
070                        this.transIn = transIn;
071                        this.transOut = transOut;
072                        this.depth = (parentElement != null) ? parentElement.depth+1 : 0;
073                }
074        }
075
076        public MealyDHC(Alphabet<I> alphabet, MembershipOracle<I, Word<O>> oracle) {
077                this.alphabet = alphabet;
078                this.oracle = oracle;
079                this.suffixFinder = GlobalSuffixFinders.RIVEST_SCHAPIRE;
080        }
081
082        @Override
083        public void startLearning() {
084
085                // the effective alphabet is the concatenation of the real alphabet
086                // wrapped in Words and the list of splitters
087                List<Word<I>> effectivealpha = new ArrayList<>();
088                for (I input : alphabet) {
089                        effectivealpha.add(Word.fromLetter(input));
090                }
091                effectivealpha.addAll(splitters);
092
093                // initialize structure to store state output signatures
094                Map<List<Word<O>>, Integer> signatures = new HashMap<>();
095
096                // set up new hypothesis machine
097                hypothesis = new CompactMealy<>(alphabet);
098
099                // initialize exploration queue
100                Queue<QueueElement<I,O>> queue = new LinkedList<>();
101                
102                // initialize storage for access sequences
103                accessSequences = hypothesis.createDynamicStateMapping();
104
105                // first element to be explored represents the initial state with no predecessor
106                queue.add(new QueueElement<I,O>(null, null, null, null));
107
108                Deduplicator<Word<O>> deduplicator = new Deduplicator<>();
109                
110                while (!queue.isEmpty()) {
111                        // get element to be explored from queue
112                        QueueElement<I,O> elem = queue.poll();
113
114                        // determine access sequence for state
115                        Word<I> access = assembleAccessSequence(elem);
116
117                        // assemble queries
118                        ArrayList<DefaultQuery<I, Word<O>>> queries = new ArrayList<>(effectivealpha.size());
119                        for (Word<I> suffix : effectivealpha) {
120                                queries.add(new DefaultQuery<I, Word<O>>(access, suffix));
121                        }
122
123                        // retrieve answers
124                        oracle.processQueries(queries);
125
126                        // assemble output signature
127                        List<Word<O>> sig = new ArrayList<>(effectivealpha.size());
128                        for (DefaultQuery<I, Word<O>> query : queries) {
129                                sig.add(deduplicator.deduplicate(query.getOutput()));
130                        }
131
132                        Integer sibling = signatures.get(sig);
133
134                        if (sibling != null) {
135                                // this element does not possess a new output signature
136                                // create a transition from parent state to sibling
137                                hypothesis.addTransition(elem.parentState, elem.transIn, sibling, elem.transOut);
138                        } else {
139                                // this is actually an observably distinct state! Progress!
140                                // Create state and connect via transition to parent
141                                Integer state = elem.parentElement == null ? hypothesis.addInitialState() : hypothesis.addState();
142                                if (elem.parentElement != null) {
143                                        hypothesis.addTransition(elem.parentState, elem.transIn, state, elem.transOut);
144                                }
145                                signatures.put(sig, state);
146                                accessSequences.put(state, elem);
147
148                                scheduleSuccessors(elem, state, queue, sig);
149                        }
150                }
151        }
152
153        private Word<I> assembleAccessSequence(QueueElement<I,O> elem) {
154                List<I> word = new ArrayList<>(elem.depth);
155                
156                QueueElement<I,O> pre = elem.parentElement;
157                I sym = elem.transIn;
158                while(pre != null && sym != null) {
159                        word.add(sym);
160                        sym = pre.transIn;
161                        pre = pre.parentElement;
162                }
163                
164                Collections.reverse(word);
165                return Word.fromList(word);
166        }
167
168        private void scheduleSuccessors(QueueElement<I,O> elem, Integer state, Queue<QueueElement<I,O>> queue, List<Word<O>> sig) throws IllegalArgumentException {
169                for (int i = 0; i < alphabet.size(); ++i) {
170                        // retrieve I/O for transition
171                        I input = alphabet.getSymbol(i);
172                        O output = sig.get(i).getSymbol(0);
173
174                        // create successor element and schedule for exploration
175                        queue.add(new QueueElement<>(state, elem, input, output));
176                }
177        }
178        
179        private void checkInternalState() {
180                if (hypothesis == null) {
181                        throw new IllegalStateException("No hypothesis learned yet");
182                }
183        }
184
185        @Override
186        public boolean refineHypothesis(DefaultQuery<I, Word<O>> ceQuery) {
187                checkInternalState();
188
189                int oldsize = hypothesis.size();
190                
191                for(Word<I> suf : suffixFinder.findSuffixes(ceQuery, this, hypothesis, oracle)) {
192                        if(!splitters.contains(suf)) {
193                                splitters.add(suf);
194                                log.log(Level.FINE, "added suffix: {0}", suf);
195                        }
196                }
197
198                startLearning();
199
200                return oldsize != hypothesis.size();
201        }
202
203        @Override
204        public MealyMachine<?, I, ?, O> getHypothesisModel() {
205                checkInternalState();
206                return hypothesis;
207        }
208        
209        @Override
210        public Word<I> transformAccessSequence(Word<I> word) {
211                checkInternalState();
212                Integer state = hypothesis.getSuccessor(hypothesis.getInitialState(), word);
213                return assembleAccessSequence(accessSequences.get(state));
214        }
215
216        @Override
217        public boolean isAccessSequence(Word<I> word) {
218                checkInternalState();
219                Word<I> canonical = transformAccessSequence(word);
220                return canonical.equals(word);
221        }
