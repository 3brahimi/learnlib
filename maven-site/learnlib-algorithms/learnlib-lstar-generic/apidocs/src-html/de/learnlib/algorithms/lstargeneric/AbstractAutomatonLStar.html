017package de.learnlib.algorithms.lstargeneric;
018
019import java.util.ArrayList;
020
021import net.automatalib.automata.MutableDeterministic;
022import net.automatalib.commons.util.Pair;
023import net.automatalib.commons.util.collections.CollectionsUtil;
024import net.automatalib.words.Alphabet;
025import de.learnlib.algorithms.lstargeneric.table.Row;
026import de.learnlib.api.MembershipOracle;
027
028
029/**
030 * Abstract base class for algorithms that produce (subclasses of) {@link MutableDeterministic}
031 * automata. 
032 * 
033 * This class provides the L*-style hypothesis construction. Implementing classes solely have
034 * to specify how state and transition properties should be derived.
035 *  
036 * @author Malte Isberner <malte.isberner@gmail.com>
037 *
038 * @param <A> automaton class, must be a subclass of {@link MutableDeterministic}
039 * @param <I> input symbol class
040 * @param <O> output class
041 * @param <SP> state property class
042 * @param <TP> transition property class
043 */
044public abstract class AbstractAutomatonLStar<A,I,O,S,T,SP,TP,AI extends MutableDeterministic<S,I,T,SP,TP>> extends
045                AbstractLStar<A, I, O> {
046
047        protected final AI internalHyp;
048        protected final ArrayList<Pair<Row<I>,S>> stateInfos
049                = new ArrayList<Pair<Row<I>,S>>();
050        
051        /**
052         * Constructor.
053         * @param alphabet the learning alphabet
054         * @param oracle the learning oracle
055         */
056        public AbstractAutomatonLStar(Alphabet<I> alphabet,
057                        MembershipOracle<I, O> oracle,
058                        AI internalHyp) {
059                super(alphabet, oracle);
060                this.internalHyp = internalHyp;
061                internalHyp.clear();
062        }
063        
064        /**
065         * Derives a state property from the corresponding row.
066         * @param stateRow the row for which the state is created
067         * @return the state property of the corresponding state
068         */
069        protected abstract SP stateProperty(Row<I> stateRow);
070        
071        /**
072         * Derives a transition property from the corresponding transition.
073         * N.B.: Not the transition row is passed to this method, but the
074         * row for the outgoing state. The transition row can be retrieved
075         * by <code>stateRow.getSuccessor(inputIdx)</code>.
076         * @param stateRow the row for the source state
077         * @param inputIdx the index of the input symbol to consider
078         * @return the transition property of the corresponding transition
079         */
080        protected abstract TP transitionProperty(Row<I> stateRow, int inputIdx);
081        
082        
083        protected abstract A exposeInternalHypothesis();
084        
085        /**
086         * Performs the L*-style hypothesis construction.
087         * For creating states and transitions, the {@link #stateProperty(Row)} and
088         * {@link #transitionProperty(Row, int)} methods are used to derive the
089         * respective properties.
090         * @param model the model to output
091         */
092        protected void updateInternalHypothesis() {
093                if(!table.isInitialized())
094                        throw new IllegalStateException("Cannot update internal hypothesis: not initialized");
095                
096                int oldStates = internalHyp.size();
097                int numDistinct = table.numDistinctRows();
098                
099
100                int newStates = numDistinct - oldStates;
101                
102                if(newStates <= 0)
103                        return;
104                
105                stateInfos.addAll(CollectionsUtil.<Pair<Row<I>,S>>nullList(newStates));
106                
107                
108                // TODO: Is there a quicker way than iterating over *all* rows?
109                // FIRST PASS: Create new hypothesis states
110                for(Row<I> sp : table.getShortPrefixRows()) {
111                        int id = sp.getRowContentId();
112                        Pair<Row<I>,S> info = stateInfos.get(id);
113                        if(info != null) {
114                                // State from previous hypothesis, property might have changed
115                                if(info.getFirst() == sp)
116                                        internalHyp.setStateProperty(info.getSecond(), stateProperty(sp));
117                                continue;
118                        }
119                        
120                        S state = createState((id == 0), sp);
121                        
122                        stateInfos.set(id, Pair.make(sp, state));
123                }
124                
125                // SECOND PASS: Create hypothesis transitions
126                for(Pair<Row<I>,S> info : stateInfos) {
127                        Row<I> sp = info.getFirst();
128                        int rowId = sp.getRowContentId();
129                        S state = info.getSecond();
130                        
131                        for(int i = 0; i < alphabet.size(); i++) {
132                                I input = alphabet.getSymbol(i);
133                                
134                                Row<I> succ = sp.getSuccessor(i);
135                                int succId = succ.getRowContentId();
136                                
137                                if(rowId < oldStates && succId < oldStates)
138                                        continue;
139                                
140                                S succState = stateInfos.get(succId).getSecond();
141                                
142                                setTransition(state, input, succState, sp, i, succ);
143                        }
144                }
145        }
146        
147        protected S createState(boolean initial, Row<I> row) {
148                SP prop = stateProperty(row);
149                if(initial)
150                        return internalHyp.addInitialState(prop);
151                return internalHyp.addState(prop);
152        }
153        
154        protected void setTransition(S from, I input, S to, Row<I> fromRow, int inputIdx, Row<I> toRow) {
155                TP prop = transitionProperty(fromRow, inputIdx);
156                internalHyp.setTransition(from, input, to, prop);
157        }
158        
159        @Override
160        public A getHypothesisModel() {
161                updateInternalHypothesis();
162                return exposeInternalHypothesis();
163        }
164
165}
